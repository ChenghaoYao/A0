<script>
/*
=====================================================================
D3-真实承诺 v1.94 (保留 v1.93 全部功能，增加 CSV 导出)
A: AbsCoords B: FixedFrame(深蓝空心) + LandBlocks(浅蓝空心)
C/D/E: RelCoords F: 上层叙事
=====================================================================
*/

/* ========== 模块A：绝对坐标系 ========== */
var AbsCoords = {
  voxelRenderSize: 1.0,
  snapToCorner: function(x,y,z){ return {x:Math.round(x),y:Math.round(y),z:Math.round(z)}; },
  getVoxelRenderCenter: function(vc){ return {x:vc.x*this.voxelRenderSize,y:vc.y*this.voxelRenderSize,z:vc.z*this.voxelRenderSize}; },
  worldToVoxel: function(wx,wy,wz){ return this.snapToCorner(Math.round(wx/this.voxelRenderSize),Math.round(wy/this.voxelRenderSize),Math.round(wz/this.voxelRenderSize)); },
  isSnapEqual: function(a,b){ return a.x===b.x && a.y===b.y && a.z===b.z; }
};

/* ========== 模块B：固定坐标系（深蓝色空心立方体） ========== */
var FixedFrame = {
  voxelAB:22, voxelAD:22, voxelAE:22, meterAB:3.0, meterAD:3.0, meterAE:3.0,
  showCornerLabels:false, penetrateMode:false,
  hollowBoxMeshes:[], cornerLabelSprites:[], scene:null, redMidpoint:null,
  opacityFace:0.35, opacityGrid:0.5, gridLines:null,
  initScene:function(s){ this.scene=s; },
  recalcMeters:function(){ this.meterAD=this.meterAB*(this.voxelAD/this.voxelAB); this.meterAE=this.meterAB*(this.voxelAE/this.voxelAB); },
  getVoxelMinMax:function(){
    var hx=Math.floor(this.voxelAB/2), hy=Math.floor(this.voxelAE/2), hz=Math.floor(this.voxelAD/2);
    if(this.voxelAB%2===0){ return {xMin:-hx, xMax:hx-1, yMin:-hy, yMax:this.voxelAE%2===0?hy-1:hy, zMin:-hz, zMax:this.voxelAD%2===0?hz-1:hz}; }
    else{ return {xMin:-hx, xMax:hx, yMin:-hy, yMax:hy, zMin:-hz, zMax:hz}; }
  },
  rebuildHollowBox:function(){
    if(!this.scene) return;
    for(var i=0;i<this.hollowBoxMeshes.length;i++) this.scene.remove(this.hollowBoxMeshes[i]); this.hollowBoxMeshes=[];
    for(var j=0;j<this.cornerLabelSprites.length;j++) this.scene.remove(this.cornerLabelSprites[j]); this.cornerLabelSprites=[];
    if(this.gridLines){ this.scene.remove(this.gridLines); this.gridLines=null; }
    if(this.redMidpoint){ this.scene.remove(this.redMidpoint); this.redMidpoint=null; }
    var mm=this.getVoxelMinMax(), vs=AbsCoords.voxelRenderSize;
    for(var vx=mm.xMin; vx<=mm.xMax; vx++){ for(var vy=mm.yMin; vy<=mm.yMax; vy++){ for(var vz=mm.zMin; vz<=mm.zMax; vz++){
      if(!(vx===mm.xMin||vx===mm.xMax||vy===mm.yMin||vy===mm.yMax||vz===mm.zMin||vz===mm.zMax)) continue;
      var rc=AbsCoords.getVoxelRenderCenter({x:vx,y:vy,z:vz});
      var geo=new THREE.BoxGeometry(vs*0.95, vs*0.95, vs*0.95);
      var mat=new THREE.MeshBasicMaterial({color:0x3366aa, transparent:true, opacity:0.45});
      var mesh=new THREE.Mesh(geo,mat); mesh.position.set(rc.x,rc.y,rc.z);
      mesh.userData={type:'frameBlock', vx:vx, vy:vy, vz:vz};
      mesh.add(new THREE.LineSegments(new THREE.EdgesGeometry(geo), new THREE.LineBasicMaterial({color:0x000000})));
      this.scene.add(mesh); this.hollowBoxMeshes.push(mesh);
    }}}
    var midGeo=new THREE.BoxGeometry(vs,vs,vs); this.redMidpoint=new THREE.Mesh(midGeo, new THREE.MeshBasicMaterial({color:0xff0000}));
    this.redMidpoint.position.set(0,0,0); this.scene.add(this.redMidpoint);
    this.gridLines=new THREE.Group();
    for(var ix=mm.xMin; ix<=mm.xMax+1; ix++){ var x=ix*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(x,mm.yMin*vs,mm.zMin*vs), new THREE.Vector3(x,mm.yMax*vs+vs,mm.zMin*vs), new THREE.Vector3(x,mm.yMax*vs+vs,mm.zMax*vs+vs), new THREE.Vector3(x,mm.yMin*vs,mm.zMax*vs+vs), new THREE.Vector3(x,mm.yMin*vs,mm.zMin*vs)]); this.gridLines.add(new THREE.Line(geo, new THREE.LineBasicMaterial({color:0x000000, transparent:true, opacity:this.opacityGrid}))); }
    for(var iz=mm.zMin; iz<=mm.zMax+1; iz++){ var z=iz*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(mm.xMin*vs,mm.yMin*vs,z), new THREE.Vector3(mm.xMax*vs+vs,mm.yMin*vs,z), new THREE.Vector3(mm.xMax*vs+vs,mm.yMax*vs+vs,z), new THREE.Vector3(mm.xMin*vs,mm.yMax*vs+vs,z), new THREE.Vector3(mm.xMin*vs,mm.yMin*vs,z)]); this.gridLines.add(new THREE.Line(geo, new THREE.LineBasicMaterial({color:0x000000, transparent:true, opacity:this.opacityGrid}))); }
    for(var iy=mm.yMin; iy<=mm.yMax+1; iy++){ var y=iy*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(mm.xMin*vs,y,mm.zMin*vs), new THREE.Vector3(mm.xMax*vs+vs,y,mm.zMin*vs), new THREE.Vector3(mm.xMax*vs+vs,y,mm.zMax*vs+vs), new THREE.Vector3(mm.xMin*vs,y,mm.zMax*vs+vs), new THREE.Vector3(mm.xMin*vs,y,mm.zMin*vs)]); this.gridLines.add(new THREE.Line(geo, new THREE.LineBasicMaterial({color:0x000000, transparent:true, opacity:this.opacityGrid}))); }
    this.scene.add(this.gridLines);
    this.updateCornerLabels();
  },
  updateCornerLabels:function(){
    for(var j=0;j<this.cornerLabelSprites.length;j++) this.scene.remove(this.cornerLabelSprites[j]); this.cornerLabelSprites=[];
    if(!this.showCornerLabels) return;

function parseCoordText(text){
  var tpMatch=text.match(/TP\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/); if(tpMatch){ var tx=parseFloat(tpMatch[1]), ty=parseFloat(tpMatch[2]), tz=parseFloat(tpMatch[3]); if(RelCoords.internalCamera){ RelCoords.internalCamera.position.set(tx,ty,tz); RelCoords.internalCamera.lookAt(0,0,0); statusBar.innerHTML='已传送到('+tx.toFixed(1)+','+ty.toFixed(1)+','+tz.toFixed(1)+')'; } return []; }
  var fillMatch=text.match(/\/fill\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+minecraft:(glass|dirt|ice)/i); if(fillMatch){ PathManager.executeFill(parseInt(fillMatch[1]),parseInt(fillMatch[2]),parseInt(fillMatch[3]),parseInt(fillMatch[4]),parseInt(fillMatch[5]),parseInt(fillMatch[6]),fillMatch[7].toLowerCase(),RelCoords.scene); return []; }
  var points=[]; var regex=/(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g; var match; while((match=regex.exec(text))!==null){ points.push({type:match[1], x:parseInt(match[2]), y:parseInt(match[3]), z:parseInt(match[4])}); } if(points.length===0) return []; var qd=[], zd=[], dn=[]; for(var i=0;i<points.length;i++){ if(points[i].type==='QD') qd.push(points[i]); else if(points[i].type==='ZD') zd.push(points[i]); else if(points[i].type.indexOf('D')===0) dn.push(points[i]); } dn.sort(function(a,b){ return parseInt(a.type.substring(1))-parseInt(b.type.substring(1)); }); return qd.concat(dn).concat(zd);
}

function autoFixPath(){ for(var i=0;i<PathManager.pathPoints.length;i++){ var p=PathManager.pathPoints[i]; if(PathManager.isInsideObstacle(p.x,p.y,p.z)){ var f=PathManager.findNearestValidVoxel(p.x,p.y,p.z); p.x=f.x; p.y=f.y; p.z=f.z; } } PathManager.syncPathToText(); PathManager.refreshSceneVisuals(RelCoords.scene); obstacleWarning.style.display='none'; statusBar.innerHTML='路径已自动修正'; }

  document.getElementById('exportCsvBtn').addEventListener('click', function(){
    if(PathManager.pathPoints.length < 2){
      statusBar.innerHTML = '❌ 无有效路径点，请先解析路径';
      return;
    }
    var originLat = parseFloat(document.getElementById('originLat').value);
    var originLng = parseFloat(document.getElementById('originLng').value);
    var originAlt = parseFloat(document.getElementById('originAlt').value);
    var meterPerVoxel = parseFloat(document.getElementById('meterPerVoxel').value);
    if(isNaN(originLat) || isNaN(originLng) || isNaN(originAlt) || isNaN(meterPerVoxel) || meterPerVoxel <= 0){
      statusBar.innerHTML = '❌ 请正确填写原点GPS和体素边长（>0）';
      return;
    }
    var csvRows = ['latitude,longitude,altitude(m),speed(m/s)'];
    var speed = 5.0;   // 航点速度固定 5m/s，需要可调后续再加
    for(var i=0; i<PathManager.pathPoints.length; i++){
      var p = PathManager.pathPoints[i];
      // 体素坐标 → 经纬度 (复用之前推演的公式)
      // 纬度方向: 步长 = 体素边长(米) / 111320
      // 经度方向: 步长 = 体素边长(米) / (111320 * cos(originLat弧度))
      var latStep = meterPerVoxel / 111320.0;
      var lngStep = meterPerVoxel / (111320.0 * Math.cos(originLat * Math.PI / 180.0));
      var lat = originLat + p.z * latStep;
      var lng = originLng + p.x * lngStep;
      var alt = originAlt + p.y * meterPerVoxel;
      csvRows.push(lat.toFixed(7) + ',' + lng.toFixed(7) + ',' + alt.toFixed(2) + ',' + speed);
    }
    var csvBlob = new Blob([csvRows.join('\n')], {type: 'text/csv'});
    var link = document.createElement('a');
    var url = URL.createObjectURL(csvBlob);
    link.href = url;
    link.download = 'd3_path.csv';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
    statusBar.innerHTML = '✅ 已导出 ' + (PathManager.pathPoints.length) + ' 个航点 → d3_path.csv';
  });
}
