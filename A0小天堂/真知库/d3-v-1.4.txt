/* ========== 模块0：全局状态管理 ========== */
const D3State = {
    worldSize: { ad: 4, ab: 3, ae: 2.5 },
    voxelCount: 22,
    voxelPhysicalSize: { x: 0, y: 0, z: 0 },
    pathPoints: [],
    anchorMeshes: [],
    lockTranslate: false, lockRotate: false, lockScale: false,
    opacityFace: 0.35, opacityGrid: 0.5,
    scene: null,
    externalRenderer: null, internalRenderer: null,
    externalCamera: null, internalCamera: null,
    externalControls: null,
    boxMesh: null, gridLines: null,
    redMidpoint: null,
    pathLine: null, arrowMesh: null,
    externalAnimId: null, internalAnimId: null, manhattanAnimId: null,
    // 内部窗口状态
    internalMode: 'observe',
    internalMoveForward: false, internalMoveBack: false,
    internalMoveLeft: false, internalMoveRight: false,
    internalMoveUp: false, internalMoveDown: false,
    internalMouseDown: false,
    internalLastMouseX: 0, internalLastMouseY: 0,
    internalPitch: 0, internalYaw: 0,
    internalReady: false,
    bleDevice: null, bleCharacteristic: null, wifiSocket: null,
    init() { this.calcVoxelSize(); },
    calcVoxelSize() {
        this.voxelPhysicalSize.x = this.worldSize.ab / this.voxelCount;
        this.voxelPhysicalSize.z = this.worldSize.ad / this.voxelCount;
        this.voxelPhysicalSize.y = this.worldSize.ae / this.voxelCount;
    },
    snapToCorner(x, y, z) {
        const sx = Math.round(x / this.voxelPhysicalSize.x) * this.voxelPhysicalSize.x;
        const sy = Math.round(y / this.voxelPhysicalSize.y) * this.voxelPhysicalSize.y;
        const sz = Math.round(z / this.voxelPhysicalSize.z) * this.voxelPhysicalSize.z;
        return { x: parseFloat(sx.toFixed(4)), y: parseFloat(sy.toFixed(4)), z: parseFloat(sz.toFixed(4)) };
    }
};
D3State.init();

/* ========== 模块6：文本输入与路径管理 ========== */
const coordInput = document.getElementById('coordInput');

function parseCoordText(text) {
    const tpMatch = text.match(/TP\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/);
    if (tpMatch) {
        const tx = parseFloat(tpMatch[1]), ty = parseFloat(tpMatch[2]), tz = parseFloat(tpMatch[3]);
        if (D3State.internalCamera) {
            D3State.internalCamera.position.set(tx, ty, tz);
            statusBar.innerHTML = `✅ 已传送到 (${tx}, ${ty}, ${tz})`;
        }
        return [];
    }
    const points = [];
    const regex = /(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g;
    let match;
    while ((match = regex.exec(text)) !== null) {
        points.push({ type: match[1], x: parseFloat(match[2]), y: parseFloat(match[3]), z: parseFloat(match[4]) });
    }
    if (points.length === 0) return [];
    const qd = points.filter(p => p.type === 'QD');
    const zd = points.filter(p => p.type === 'ZD');
    let dn = points.filter(p => p.type.startsWith('D') && p.type !== 'QD' && p.type !== 'ZD');
    dn.sort((a, b) => (parseInt(a.type.substring(1)) - parseInt(b.type.substring(1))));
    return [...qd, ...dn, ...zd].map(p => {
        const s = D3State.snapToCorner(p.x, p.y, p.z);
        return { type: p.type, x: s.x, y: s.y, z: s.z };
    });
}

function generatePackage(points) {
    if (points.length < 2) return null;
    let out = points.length === 2 ? '路径类型：直线路径\n' : '路径类型：自由路径\n';
    out += `总段数：${points.length - 1}\n\n路径坐标清单：\n`;
    out += points.map(p => `${p.type}(${p.x},${p.y},${p.z})`).join(' → ') + '\n\n';
    for (let i = 0; i < points.length - 1; i++) {
        const a = points[i], b = points[i+1];
        out += `段${i+1}: legX=${Math.abs(b.x-a.x).toFixed(2)}m, legZ=${Math.abs(b.z-a.z).toFixed(2)}m, legY=${Math.abs(b.y-a.y).toFixed(2)}m\n`;
    }
    return out;
}

function updateAnchorsInScene() {
    if (!D3State.scene) return;
    D3State.anchorMeshes.forEach(m => D3State.scene.remove(m));
    D3State.anchorMeshes = [];
    D3State.pathPoints.forEach(p => {
        const geo = new THREE.BoxGeometry(D3State.voxelPhysicalSize.x*0.8, D3State.voxelPhysicalSize.y*0.8, D3State.voxelPhysicalSize.z*0.8);
        let color = p.type === 'QD' ? 0x00ff00 : p.type === 'ZD' ? 0x0088ff : 0xffffff;
        const mesh = new THREE.Mesh(geo, new THREE.MeshBasicMaterial({ color }));
        mesh.position.set(p.x, p.y, p.z);
        D3State.scene.add(mesh);
        D3State.anchorMeshes.push(mesh);
    });
}

function updatePathLineInScene() {
    if (!D3State.scene) return;
    if (D3State.pathLine) D3State.scene.remove(D3State.pathLine);
    if (D3State.arrowMesh) D3State.scene.remove(D3State.arrowMesh);
    if (D3State.pathPoints.length < 2) return;
    const pts = D3State.pathPoints.map(p => new THREE.Vector3(p.x, p.y, p.z));
    const geo = new THREE.BufferGeometry().setFromPoints(pts);
    D3State.pathLine = new THREE.Line(geo, new THREE.LineBasicMaterial({ color: 0xff3333, linewidth: 2 }));
    D3State.scene.add(D3State.pathLine);
    const last = pts[pts.length-1], prev = pts[pts.length-2] || pts[0];
    const dir = new THREE.Vector3().subVectors(last, prev).normalize();
    if (dir.length() < 0.001) dir.set(0,0,1);
    const arrowMesh = new THREE.Mesh(new THREE.ConeGeometry(0.15,0.4,8,8), new THREE.MeshBasicMaterial({ color: 0xff3333 }));
    arrowMesh.position.copy(last);
    arrowMesh.quaternion.setFromUnitVectors(new THREE.Vector3(0,1,0), dir);
    D3State.arrowMesh = arrowMesh;
    D3State.scene.add(arrowMesh);
}

function updateMidpointMarker() {
    if (!D3State.scene || !D3State.redMidpoint) return;
    const s = D3State.snapToCorner(0,0,0);
    D3State.redMidpoint.position.set(s.x, s.y, s.z);
    D3State.redMidpoint.visible = true;
}

function refreshSceneVisuals() { updateAnchorsInScene(); updatePathLineInScene(); updateMidpointMarker(); }

    }
    D3State.pathPoints = points;
    refreshSceneVisuals();
    statusBar.innerHTML = `✅ 路径已更新 | ${points.length}个点`;
    outputArea.textContent = '路径已更新，点击"生成指令包"查看详情。';
});

function startRenderLoop(name) {
    if (name === 'external' && !activeRenderLoops.external && D3State.externalRenderer && D3State.externalCamera) {
        activeRenderLoops.external = true;
        function r(){ if(!activeRenderLoops.external) return; D3State.externalRenderer.render(D3State.scene, D3State.externalCamera); D3State.externalAnimId = requestAnimationFrame(r); }
        r();
    }
    if (name === 'internal' && !activeRenderLoops.internal && D3State.internalRenderer && D3State.internalCamera) {

            statusBar.innerHTML = '⏳ 内部窗口正在初始化，请稍后再试...';

        r();

}

function drawManhattanWindow() {
    const canvas = document.getElementById('manhattanCanvas');
    if (!canvas) return;
    const container = document.getElementById('manhattanPanel');
    const w = document.getElementById('windowContainer').clientWidth;
    const h = document.getElementById('windowContainer').clientHeight;
    canvas.width = w; canvas.height = h;
    const ctx = canvas.getContext('2d');
    ctx.clearRect(0,0,w,h);
    ctx.fillStyle = '#121420'; ctx.fillRect(0,0,w,h);
    const gs = 20, cols = Math.floor(w/gs), rows = Math.floor(h/gs);
    ctx.strokeStyle = '#333'; ctx.lineWidth = 0.5;
    for (let i=0;i<=cols;i++) { ctx.beginPath(); ctx.moveTo(i*gs,0); ctx.lineTo(i*gs,rows*gs); ctx.stroke(); }
    for (let i=0;i<=rows;i++) { ctx.beginPath(); ctx.moveTo(0,i*gs); ctx.lineTo(cols*gs,i*gs); ctx.stroke(); }
    if (D3State.pathPoints.length > 0) {
        ctx.strokeStyle = '#ff3333'; ctx.lineWidth = 2; ctx.beginPath();
        const sx = w/D3State.worldSize.ab, sz = h/D3State.worldSize.ad;
        ctx.moveTo(D3State.pathPoints[0].x*sx, D3State.pathPoints[0].z*sz);
        for (let i=1;i<D3State.pathPoints.length;i++) ctx.lineTo(D3State.pathPoints[i].x*sx, D3State.pathPoints[i].z*sz);
        ctx.stroke();
    }
    ctx.fillStyle = '#fff'; ctx.font = '12px monospace'; ctx.fillText('二维曼哈顿俯视图', 10, 20);
}

function updateInternalMovement() {
    if (!D3State.internalCamera) return;
    const speed = 0.1;
    const forward = new THREE.Vector3(0,0,-1).applyQuaternion(D3State.internalCamera.quaternion);
    forward.y = 0; forward.normalize();
    const right = new THREE.Vector3(1,0,0).applyQuaternion(D3State.internalCamera.quaternion);
    right.y = 0; right.normalize();
    if (D3State.internalMoveForward) D3State.internalCamera.position.addScaledVector(forward, speed);
    if (D3State.internalMoveBack) D3State.internalCamera.position.addScaledVector(forward, -speed);
    if (D3State.internalMoveRight) D3State.internalCamera.position.addScaledVector(right, speed);
    if (D3State.internalMoveLeft) D3State.internalCamera.position.addScaledVector(right, -speed);
    if (D3State.internalMoveUp) D3State.internalCamera.position.y += speed;
    if (D3State.internalMoveDown) D3State.internalCamera.position.y -= speed;
    document.getElementById('internalCoords').textContent =
        `${D3State.internalCamera.position.x.toFixed(2)}, ${D3State.internalCamera.position.y.toFixed(2)}, ${D3State.internalCamera.position.z.toFixed(2)}`;
}

function setupInternalControls() {
    const intPanel = document.getElementById('internalPanel');
    window.addEventListener('keydown', (e) => {
        if (!document.getElementById('internalPanel').classList.contains('active')) return;
        switch(e.key.toLowerCase()) {
            case 'w': D3State.internalMoveForward = true; e.preventDefault(); break;
            case 's': D3State.internalMoveBack = true; e.preventDefault(); break;
            case 'a': D3State.internalMoveLeft = true; e.preventDefault(); break;
            case 'd': D3State.internalMoveRight = true; e.preventDefault(); break;
            case 'q': D3State.internalMoveDown = true; e.preventDefault(); break;
            case 'e': D3State.internalMoveUp = true; e.preventDefault(); break;
        }
    });
    window.addEventListener('keyup', (e) => {
        switch(e.key.toLowerCase()) {
            case 'w': D3State.internalMoveForward = false; e.preventDefault(); break;
            case 's': D3State.internalMoveBack = false; e.preventDefault(); break;
            case 'a': D3State.internalMoveLeft = false; e.preventDefault(); break;
            case 'd': D3State.internalMoveRight = false; e.preventDefault(); break;
            case 'q': D3State.internalMoveDown = false; e.preventDefault(); break;
            case 'e': D3State.internalMoveUp = false; e.preventDefault(); break;
        }
    });
    const modeIndicator = document.getElementById('internalMode');
    window.addEventListener('keydown', (e) => {
        if (!document.getElementById('internalPanel').classList.contains('active')) return;
        if (e.key === '1') {
            D3State.internalMode = 'place';
            modeIndicator.textContent = '🧱 放置模式';
            e.preventDefault();
        } else if (e.key === '0') {
            D3State.internalMode = 'remove';
            modeIndicator.textContent = '🗑️ 拆除模式';
            e.preventDefault();
        } else if (e.key === 'Escape') {
            D3State.internalMode = 'observe';
            modeIndicator.textContent = '👁️ 观察模式';
            e.preventDefault();
        }
    });
    intPanel.addEventListener('pointerdown', (e) => {
        D3State.internalMouseDown = true;
        D3State.internalLastMouseX = e.clientX;
        D3State.internalLastMouseY = e.clientY;
        intPanel.setPointerCapture(e.pointerId);
        e.preventDefault();
    });
    window.addEventListener('pointermove', (e) => {
        if (!D3State.internalMouseDown) return;
        if (!document.getElementById('internalPanel').classList.contains('active')) return;
        const dx = e.clientX - D3State.internalLastMouseX;
        const dy = e.clientY - D3State.internalLastMouseY;
        D3State.internalLastMouseX = e.clientX;
        D3State.internalLastMouseY = e.clientY;
        D3State.internalYaw -= dx * 0.002;
        D3State.internalPitch -= dy * 0.002;
        D3State.internalPitch = Math.max(-Math.PI/2, Math.min(Math.PI/2, D3State.internalPitch));
        const q = new THREE.Quaternion();
        const euler = new THREE.Euler(D3State.internalPitch, D3State.internalYaw, 0, 'YXZ');
        q.setFromEuler(euler);
        D3State.internalCamera.quaternion.copy(q);
    });
    window.addEventListener('pointerup', () => { D3State.internalMouseDown = false; });
    intPanel.addEventListener('click', (e) => {
        if (D3State.internalMode === 'observe') return;
        if (!D3State.scene) return;
        const raycaster = new THREE.Raycaster();
        const mouse = new THREE.Vector2();
        const rect = intPanel.getBoundingClientRect();
        mouse.x = ((e.clientX - rect.left) / rect.width) * 2 - 1;
        mouse.y = -((e.clientY - rect.top) / rect.height) * 2 + 1;
        raycaster.setFromCamera(mouse, D3State.internalCamera);
        const intersects = raycaster.intersectObjects([D3State.boxMesh], false);
        if (intersects.length > 0) {
            const point = intersects[0].point;
            const snapped = D3State.snapToCorner(point.x, point.y, point.z);
            if (D3State.internalMode === 'place') {
                const geo = new THREE.BoxGeometry(D3State.voxelPhysicalSize.x*0.9, D3State.voxelPhysicalSize.y*0.9, D3State.voxelPhysicalSize.z*0.9);
                const mat = new THREE.MeshBasicMaterial({ color: 0xffffff });
                const cube = new THREE.Mesh(geo, mat);
                cube.position.set(snapped.x, snapped.y, snapped.z);
                D3State.scene.add(cube);
                D3State.anchorMeshes.push(cube);
                statusBar.innerHTML = `🧱 放置体素 @ (${snapped.x.toFixed(2)}, ${snapped.y.toFixed(2)}, ${snapped.z.toFixed(2)})`;
            } else if (D3State.internalMode === 'remove') {
                const toRemove = [];
                D3State.scene.children.forEach(child => {
                    if (child.isMesh && child.geometry.type === 'BoxGeometry' && child !== D3State.boxMesh && child !== D3State.redMidpoint) {
                        const dist = child.position.distanceTo(new THREE.Vector3(snapped.x, snapped.y, snapped.z));
                        if (dist < D3State.voxelPhysicalSize.x * 0.6) toRemove.push(child);
                    }
                });
                toRemove.forEach(c => D3State.scene.remove(c));
                if (toRemove.length > 0) statusBar.innerHTML = `🗑️ 拆除 ${toRemove.length} 个体素`;
            }
        }
    });
}

function rebuildBoxAndGrid() {
    if (!D3State.scene) return;
    if (D3State.boxMesh) D3State.scene.remove(D3State.boxMesh);
    if (D3State.gridLines) D3State.scene.remove(D3State.gridLines);
    if (D3State.redMidpoint) D3State.scene.remove(D3State.redMidpoint);
    const { ab, ad, ae } = D3State.worldSize;
    const n = D3State.voxelCount;
    D3State.boxMesh = new THREE.Mesh(new THREE.BoxGeometry(ab, ae, ad),
        new THREE.MeshBasicMaterial({ color: 0x4682c8, transparent: true, opacity: D3State.opacityFace, side: THREE.DoubleSide }));
    D3State.scene.add(D3State.boxMesh);
    const gridGroup = new THREE.Group();
    const vs = D3State.voxelPhysicalSize;
    for (let ix=0;ix<=n;ix++) {
        const x = -ab/2 + ix*vs.x;
        const geo = new THREE.BufferGeometry().setFromPoints([
            new THREE.Vector3(x,-ae/2,-ad/2),new THREE.Vector3(x,ae/2,-ad/2),new THREE.Vector3(x,ae/2,ad/2),new THREE.Vector3(x,-ae/2,ad/2),new THREE.Vector3(x,-ae/2,-ad/2)]);
        gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color:0x000000, transparent:true, opacity:D3State.opacityGrid })));
    }
    for (let iz=0;iz<=n;iz++) {
        const z = -ad/2 + iz*vs.z;
        const geo = new THREE.BufferGeometry().setFromPoints([
            new THREE.Vector3(-ab/2,-ae/2,z),new THREE.Vector3(ab/2,-ae/2,z),new THREE.Vector3(ab/2,ae/2,z),new THREE.Vector3(-ab/2,ae/2,z),new THREE.Vector3(-ab/2,-ae/2,z)]);
        gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color:0x000000, transparent:true, opacity:D3State.opacityGrid })));
    }
    for (let iy=0;iy<=n;iy++) {
        const y = -ae/2 + iy*vs.y;
        const geo = new THREE.BufferGeometry().setFromPoints([
            new THREE.Vector3(-ab/2,y,-ad/2),new THREE.Vector3(ab/2,y,-ad/2),new THREE.Vector3(ab/2,y,ad/2),new THREE.Vector3(-ab/2,y,ad/2),new THREE.Vector3(-ab/2,y,-ad/2)]);
        gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color:0x000000, transparent:true, opacity:D3State.opacityGrid })));
    }
    D3State.gridLines = gridGroup; D3State.scene.add(gridGroup);
    const midGeo = new THREE.BoxGeometry(vs.x*0.6, vs.y*0.6, vs.z*0.6);
    D3State.redMidpoint = new THREE.Mesh(midGeo, new THREE.MeshBasicMaterial({ color:0xff0000 }));
    D3State.scene.add(D3State.redMidpoint);
    updateMidpointMarker();
}

function createOrbitControls(camera, domElement) {
    let isDown=false, isRightDrag=false, lastX=0, lastY=0;
    const target = new THREE.Vector3(0,0,0);
    let spherical = new THREE.Spherical().setFromVector3(new THREE.Vector3().copy(camera.position).sub(target));
    domElement.addEventListener('pointerdown', (e) => { isRightDrag = (e.button===2||e.button===1); isDown=true; lastX=e.clientX; lastY=e.clientY; e.preventDefault(); });
    window.addEventListener('pointermove', (e) => {
        if (!isDown) return;
        const dx=e.clientX-lastX, dy=e.clientY-lastY; lastX=e.clientX; lastY=e.clientY;
        if (isRightDrag) {
            if (D3State.lockTranslate) return;
            const right = new THREE.Vector3().crossVectors(camera.up, new THREE.Vector3().copy(camera.position).normalize()).normalize();
            const up = camera.up.clone().normalize();
            target.add(right.multiplyScalar(-dx*0.01)).add(up.multiplyScalar(dy*0.01));
        } else {
            if (D3State.lockRotate) return;
            spherical.theta -= dx*0.005; spherical.phi -= dy*0.005;
            spherical.phi = Math.max(0.1, Math.min(Math.PI-0.1, spherical.phi));
        }
        const offset = new THREE.Vector3().setFromSpherical(spherical);
        camera.position.copy(target).add(offset); camera.lookAt(target);
    });
    window.addEventListener('pointerup', () => { isDown=false; isRightDrag=false; });
    domElement.addEventListener('wheel', (e) => {
        if (D3State.lockScale) return;
        spherical.radius *= (e.deltaY>0)?1.05:0.95;
        spherical.radius = Math.max(1, Math.min(20, spherical.radius));
        const offset = new THREE.Vector3().setFromSpherical(spherical);
        camera.position.copy(target).add(offset); camera.lookAt(target);
        e.preventDefault();
    });
    domElement.addEventListener('contextmenu', (e) => e.preventDefault());
    return { target, update: () => { spherical.setFromVector3(new THREE.Vector3().copy(camera.position).sub(target)); } };
}
