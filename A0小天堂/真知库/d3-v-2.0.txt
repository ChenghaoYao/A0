function parseCoordText(text){
    var tpMatch=text.match(/TP\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/);
    if(tpMatch){ var tx=parseFloat(tpMatch[1]),ty=parseFloat(tpMatch[2]),tz=parseFloat(tpMatch[3]); if(D3State.internalCamera){ D3State.internalCamera.position.set(tx,ty,tz); D3State.internalCamera.lookAt(0,0,0); statusBar.innerHTML='已传送到('+tx.toFixed(1)+','+ty.toFixed(1)+','+tz.toFixed(1)+')'; } return[]; }
    var points=[]; var regex=/(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g; var match;
    while((match=regex.exec(text))!==null){ points.push({type:match[1],x:parseInt(match[2]),y:parseInt(match[3]),z:parseInt(match[4])}); }
    if(points.length===0) return[];
    var qd=[],zd=[],dn=[];
    for(var i=0;i<points.length;i++){ if(points[i].type==='QD') qd.push(points[i]); else if(points[i].type==='ZD') zd.push(points[i]); else if(points[i].type.indexOf('D')===0) dn.push(points[i]); }
    dn.sort(function(a,b){ return parseInt(a.type.substring(1))-parseInt(b.type.substring(1)); });
    return qd.concat(dn).concat(zd);
}

function generatePackage(points){
    if(points.length<2) return null;
    var out='路径类型：'+(points.length===2?'直线':'自由')+'路径'+(D3State.closedLoopMode?'（闭环）':'（开环）')+'\n总段数：'+(points.length-1)+'\n\n路径坐标清单（体素）：\n';
    for(var i=0;i<points.length;i++) out+=points[i].type+'('+points[i].x+','+points[i].y+','+points[i].z+') → ';
    out=out.slice(0,-3)+'\n\n物理距离清单：\n'; var mpv=D3State.meterAB/D3State.voxelCountAB;
    for(var j=0;j<points.length-1;j++){ var a=points[j],b=points[j+1]; out+='段'+(j+1)+': legX='+(Math.abs(b.x-a.x)*mpv).toFixed(2)+'m, legZ='+(Math.abs(b.z-a.z)*mpv).toFixed(2)+'m, legY='+(Math.abs(b.y-a.y)*mpv).toFixed(2)+'m\n'; }
    return out;
}
function generateSequence(points){
    if(points.length<2) return null; var out='时序指令清单：\n'; var mpv=D3State.meterAB/D3State.voxelCountAB; var currentDir='';
    for(var j=0;j<points.length-1;j++){ var a=points[j],b=points[j+1]; var dx=b.x-a.x,dz=b.z-a.z; var dist=Math.sqrt(dx*dx+dz*dz)*mpv; var dir=''; if(dx>0) dir='东'; else if(dx<0) dir='西'; if(dz>0) dir+='南'; else if(dz<0) dir+='北'; if(dir==='') dir='原地'; var turn=''; if(currentDir!==''&&currentDir!==dir) turn='右转 90° | '; currentDir=dir; out+=(j+1)+'. '+turn+'前进 '+dist.toFixed(2)+'m（向'+dir+'）\n'; }
    out+=points.length+'. 停止'; return out;
}

}
function updatePathLineInScene(){
    if(!D3State.scene) return;
    if(D3State.pathLine){ D3State.scene.remove(D3State.pathLine); D3State.pathLine=null; }
    if(D3State.arrowMesh){ D3State.scene.remove(D3State.arrowMesh); D3State.arrowMesh=null; }
    if(D3State.pathPoints.length<2) return;
    var pts=[];
    for(var i=0;i<D3State.pathPoints.length;i++){ var rc=D3State.getVoxelRenderCenter(D3State.pathPoints[i]); pts.push(new THREE.Vector3(rc.x,rc.y,rc.z)); }
    if(D3State.closedLoopMode&&D3State.pathPoints.length>=2){ var firstRc=D3State.getVoxelRenderCenter(D3State.pathPoints[0]); pts.push(new THREE.Vector3(firstRc.x,firstRc.y,firstRc.z)); }
    var geo=new THREE.BufferGeometry().setFromPoints(pts);
    D3State.pathLine=new THREE.Line(geo,new THREE.LineBasicMaterial({color:0xff3333,linewidth:2})); D3State.scene.add(D3State.pathLine);
    var last=pts[pts.length-1],prev=pts[pts.length-2]||pts[0]; var dir=new THREE.Vector3().subVectors(last,prev).normalize(); if(dir.length()<0.001) dir.set(0,0,1);
    var arrowMesh=new THREE.Mesh(new THREE.ConeGeometry(0.2,0.5,8,8),new THREE.MeshBasicMaterial({color:0xff3333}));
    arrowMesh.position.copy(last); arrowMesh.quaternion.setFromUnitVectors(new THREE.Vector3(0,1,0),dir);
    D3State.arrowMesh=arrowMesh; D3State.scene.add(arrowMesh);
}

function addManualPoint(vc){
    for(var i=0;i<D3State.pathPoints.length;i++){ if(D3State.isSnapEqual(D3State.pathPoints[i],vc)){ statusBar.innerHTML='该位置已存在路径点'; return; } }
    var type=D3State.pathPoints.length===0?'QD':'D'+(D3State.pathPoints.filter(function(p){return p.type.indexOf('D')===0;}).length+2);
    D3State.pathPoints.push({type:type,x:vc.x,y:vc.y,z:vc.z}); D3State.syncPathToText(); refreshSceneVisuals();
    statusBar.innerHTML='放置 '+type+' @ ('+vc.x+','+vc.y+','+vc.z+')';
}
function removePointByVoxel(vc){

    if(idx<0){ statusBar.innerHTML='未找到对应路径点'; return; } collapseAfterIndex(idx-1); statusBar.innerHTML='拆除并崩塌后续路径';
}
function markAsEndpoint(vc){

function drawManhattanWindow(){
    var canvas=document.getElementById('manhattanCanvas'); if(!canvas) return;
    var w=document.getElementById('windowContainer').clientWidth, h=document.getElementById('windowContainer').clientHeight;
    canvas.width=w; canvas.height=h; var ctx=canvas.getContext('2d');
    ctx.clearRect(0,0,w,h); ctx.fillStyle='#121420'; ctx.fillRect(0,0,w,h);
    var gs=15, cols=Math.floor(w/gs), rows=Math.floor(h/gs);
    ctx.strokeStyle='#333'; ctx.lineWidth=0.5;
    for(var i=0;i<=cols;i++){ ctx.beginPath(); ctx.moveTo(i*gs,0); ctx.lineTo(i*gs,rows*gs); ctx.stroke(); }
    for(var j=0;j<=rows;j++){ ctx.beginPath(); ctx.moveTo(0,j*gs); ctx.lineTo(cols*gs,j*gs); ctx.stroke(); }
    if(D3State.pathPoints.length>0){
        ctx.strokeStyle='#ff3333'; ctx.lineWidth=2; ctx.beginPath();
        var sx=w/D3State.voxelCountAB, sz=h/D3State.voxelCountAD;
        var offsetX=w/2, offsetY=h/2;
        ctx.moveTo(offsetX+D3State.pathPoints[0].x*sx, offsetY+D3State.pathPoints[0].z*sz);
        for(var k=1;k<D3State.pathPoints.length;k++) ctx.lineTo(offsetX+D3State.pathPoints[k].x*sx, offsetY+D3State.pathPoints[k].z*sz);
        ctx.stroke();
    }
    ctx.fillStyle='#fff'; ctx.font='12px monospace'; ctx.fillText('二维曼哈顿俯视图',10,20);
}
function updateInternalMovement(){
    if(!D3State.internalCamera) return;
    var deltaWorld=0.15*D3State.voxelRenderSize;
    var forward=new THREE.Vector3(0,0,-1).applyQuaternion(D3State.internalCamera.quaternion); forward.y=0; forward.normalize();
    var right=new THREE.Vector3(1,0,0).applyQuaternion(D3State.internalCamera.quaternion); right.y=0; right.normalize();
    if(D3State.internalMoveForward) D3State.internalCamera.position.addScaledVector(forward,deltaWorld);
    if(D3State.internalMoveBack) D3State.internalCamera.position.addScaledVector(forward,-deltaWorld);
    if(D3State.internalMoveRight) D3State.internalCamera.position.addScaledVector(right,deltaWorld);
    if(D3State.internalMoveLeft) D3State.internalCamera.position.addScaledVector(right,-deltaWorld);
    if(D3State.internalMoveUp) D3State.internalCamera.position.y+=deltaWorld;
    if(D3State.internalMoveDown) D3State.internalCamera.position.y-=deltaWorld;
    var coordsEl=document.getElementById('internalCoords');
    if(coordsEl){ var ac=D3State.getCameraAbsoluteCoords(); coordsEl.textContent=ac.x.toFixed(1)+', '+ac.y.toFixed(1)+', '+ac.z.toFixed(1); }
    // 射线模式
    if(D3State.rayMode && D3State.internalCamera){
        var A=D3State.internalCamera.position;
        var dir=new THREE.Vector3(0,0,-1).applyQuaternion(D3State.internalCamera.quaternion); dir.normalize();
        var D=parseFloat(document.getElementById('rayDistance')?.value||10);
        var B_world=A.clone().add(dir.multiplyScalar(D));
        var B_voxel=D3State.worldToVoxel(B_world.x,B_world.y,B_world.z);
        if(D3State.pathPoints.length===0){
            D3State.pathPoints.push({type:'QD',x:B_voxel.x,y:B_voxel.y,z:B_voxel.z});
            D3State.pathPoints.push({type:'ZD',x:B_voxel.x,y:B_voxel.y,z:B_voxel.z});
        } else if(D3State.pathPoints.length===1){
            D3State.pathPoints.push({type:'ZD',x:B_voxel.x,y:B_voxel.y,z:B_voxel.z});
        } else {
            D3State.pathPoints[D3State.pathPoints.length-1]={type:'ZD',x:B_voxel.x,y:B_voxel.y,z:B_voxel.z};
        }
        D3State.syncPathToText(); refreshSceneVisuals();
    }
}
function startDogAtPoint(idx){
    if(D3State.dogTimer) clearInterval(D3State.dogTimer);
    D3State.dogCurrentIdx=idx;
    D3State.dogRunning=true;
    function highlightNext(){
        if(!D3State.dogRunning || D3State.dogCurrentIdx>=D3State.pathPoints.length){
            if(D3State.dogMesh) D3State.dogMesh.material.color.setHex(0xffaa00);
            if(D3State.dogTimer) clearInterval(D3State.dogTimer);
            D3State.dogRunning=false;
            return;
        }
        var p=D3State.pathPoints[D3State.dogCurrentIdx];
        var rc=D3State.getVoxelRenderCenter(p);
        D3State.dogMesh.position.set(rc.x,rc.y,rc.z);
        D3State.dogMesh.material.color.setHex(0x00ffff);
        setTimeout(function(){ if(D3State.dogMesh) D3State.dogMesh.material.color.setHex(0xffaa00); }, 200);
        D3State.dogCurrentIdx++;
    }
    D3State.dogTimer=setInterval(highlightNext, 500);
}

function rebuildBoxAndGrid(){
    if(!D3State.scene) return;
    if(D3State.boxMesh) D3State.scene.remove(D3State.boxMesh);
    if(D3State.gridLines) D3State.scene.remove(D3State.gridLines);
    if(D3State.redMidpoint) D3State.scene.remove(D3State.redMidpoint);
    var bw=D3State.voxelCountAB*D3State.voxelRenderSize, bh=D3State.voxelCountAE*D3State.voxelRenderSize, bd=D3State.voxelCountAD*D3State.voxelRenderSize;
    D3State.boxMesh=new THREE.Mesh(new THREE.BoxGeometry(bw,bh,bd),new THREE.MeshBasicMaterial({color:0x4682c8,transparent:true,opacity:D3State.opacityFace,side:THREE.DoubleSide}));
    D3State.scene.add(D3State.boxMesh);
    var gridGroup=new THREE.Group(); var vs=D3State.voxelRenderSize; var mm=D3State.getVoxelMinMax();
    for(var ix=mm.xMin;ix<=mm.xMax+1;ix++){ var x=ix*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(x,mm.yMin*vs,mm.zMin*vs),new THREE.Vector3(x,mm.yMax*vs+vs,mm.zMin*vs),new THREE.Vector3(x,mm.yMax*vs+vs,mm.zMax*vs+vs),new THREE.Vector3(x,mm.yMin*vs,mm.zMax*vs+vs),new THREE.Vector3(x,mm.yMin*vs,mm.zMin*vs)]); gridGroup.add(new THREE.Line(geo,new THREE.LineBasicMaterial({color:0x000000,transparent:true,opacity:D3State.opacityGrid}))); }
    for(var iz=mm.zMin;iz<=mm.zMax+1;iz++){ var z=iz*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(mm.xMin*vs,mm.yMin*vs,z),new THREE.Vector3(mm.xMax*vs+vs,mm.yMin*vs,z),new THREE.Vector3(mm.xMax*vs+vs,mm.yMax*vs+vs,z),new THREE.Vector3(mm.xMin*vs,mm.yMax*vs+vs,z),new THREE.Vector3(mm.xMin*vs,mm.yMin*vs,z)]); gridGroup.add(new THREE.Line(geo,new THREE.LineBasicMaterial({color:0x000000,transparent:true,opacity:D3State.opacityGrid}))); }
    for(var iy=mm.yMin;iy<=mm.yMax+1;iy++){ var y=iy*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(mm.xMin*vs,y,mm.zMin*vs),new THREE.Vector3(mm.xMax*vs+vs,y,mm.zMin*vs),new THREE.Vector3(mm.xMax*vs+vs,y,mm.zMax*vs+vs),new THREE.Vector3(mm.xMin*vs,y,mm.zMax*vs+vs),new THREE.Vector3(mm.xMin*vs,y,mm.zMin*vs)]); gridGroup.add(new THREE.Line(geo,new THREE.LineBasicMaterial({color:0x000000,transparent:true,opacity:D3State.opacityGrid}))); }

    updateMidpointMarker();
}
function createOrbitControls(camera,domElement){
    var isDown=false,isRightDrag=false,lastX=0,lastY=0; var target=new THREE.Vector3(0,0,0);
    var spherical=new THREE.Spherical().setFromVector3(new THREE.Vector3().copy(camera.position).sub(target));
    domElement.addEventListener('pointerdown',function(e){ isRightDrag=(e.button===2||e.button===1); isDown=true; lastX=e.clientX; lastY=e.clientY; e.preventDefault(); });
    window.addEventListener('pointermove',function(e){ if(!isDown) return; var dx=e.clientX-lastX,dy=e.clientY-lastY; lastX=e.clientX; lastY=e.clientY; if(isRightDrag){ if(D3State.lockTranslate) return; var right=new THREE.Vector3().crossVectors(camera.up,new THREE.Vector3().copy(camera.position).normalize()).normalize(); var up=camera.up.clone().normalize(); target.add(right.multiplyScalar(-dx*0.01)).add(up.multiplyScalar(dy*0.01)); }else{ if(D3State.lockRotate) return; spherical.theta-=dx*0.005; spherical.phi-=dy*0.005; spherical.phi=Math.max(0.1,Math.min(Math.PI-0.1,spherical.phi)); } var offset=new THREE.Vector3().setFromSpherical(spherical); camera.position.copy(target).add(offset); camera.lookAt(target); });
    window.addEventListener('pointerup',function(){ isDown=false; isRightDrag=false; });
    domElement.addEventListener('wheel',function(e){ if(D3State.lockScale) return; spherical.radius*=(e.deltaY>0)?1.05:0.95; spherical.radius=Math.max(0.05,Math.min(100,spherical.radius)); var offset=new THREE.Vector3().setFromSpherical(spherical); camera.position.copy(target).add(offset); camera.lookAt(target); e.preventDefault(); });
    domElement.addEventListener('contextmenu',function(e){ e.preventDefault(); });
    return {target:target,update:function(){ spherical.setFromVector3(new THREE.Vector3().copy(camera.position).sub(target)); }};
}

    window.addEventListener('pointerup',function(){ D3State.internalMouseDown=false; });
    intPanel.addEventListener('click',function(e){
        if(D3State.internalMode==='observe' && !D3State.structMode) return;
        if(!D3State.scene||!D3State.boxMesh) return;
        var raycaster=new THREE.Raycaster(); var mouse=new THREE.Vector2();
        var rect=intPanel.getBoundingClientRect();
        mouse.x=((e.clientX-rect.left)/rect.width)*2-1; mouse.y=-((e.clientY-rect.top)/rect.height)*2+1;
        raycaster.setFromCamera(mouse,D3State.internalCamera);
        var allObjects=[D3State.boxMesh];
        for(var i=0;i<D3State.obstacleMeshes.length;i++) allObjects.push(D3State.obstacleMeshes[i]);
        for(var j=0;j<D3State.landBlocks.length;j++){ for(var k=0;k<D3State.landBlocks[j].length;k++) allObjects.push(D3State.landBlocks[j][k]); }
        var intersects=raycaster.intersectObjects(allObjects,false);
        if(intersects.length>0){
            var hitMesh=intersects[0].object;
            var point=intersects[0].point;
            var vc=D3State.worldToVoxel(point.x,point.y,point.z);
            if(D3State.structMode){
                var structName=prompt('结构方块名称:','结构'+(D3State.structItems.length+1));
                if(structName){
                    var rc=D3State.getVoxelRenderCenter(vc);
                    var geo=new THREE.BoxGeometry(D3State.voxelRenderSize*0.95,D3State.voxelRenderSize*0.95,D3State.voxelRenderSize*0.95);
                    var mat=new THREE.MeshBasicMaterial({color:0xaa66ff,transparent:true,opacity:0.8});
                    var mesh=new THREE.Mesh(geo,mat); mesh.position.set(rc.x,rc.y,rc.z);
                    mesh.userData={type:'struct',name:structName,vx:vc.x,vy:vc.y,vz:vc.z};
                    D3State.scene.add(mesh);
                    D3State.structItems.push({name:structName,pos:{x:vc.x,y:vc.y,z:vc.z}});
                    D3State.saveStructsToStorage();
                    refreshStructList();
                    statusBar.innerHTML='结构方块已放置: '+structName;
                }
            } else if(D3State.internalMode==='place'){ addManualPoint(vc); }
            else if(D3State.internalMode==='obstacle'){ addObstaclePoint(vc); }
            else if(D3State.internalMode==='land'){ addLandPoint(vc); }
            else if(D3State.internalMode==='remove'){ removePointByVoxel(vc); }
            else if(D3State.internalMode==='endpoint'){
                var key=vc.x+','+vc.y+','+vc.z;
                D3State.endpointClickCount[key]=(D3State.endpointClickCount[key]||0)+1;
                if(D3State.endpointClickCount[key]>=3){ markAsEndpoint(vc); D3State.endpointClickCount={}; } else { statusBar.innerHTML='终点模式：已点击 '+D3State.endpointClickCount[key]+' 次，需3次确认'; }
            }
        }
    });
}
function refreshStructList(){
    var listDiv=document.getElementById('structList');
    if(!listDiv) return;
    listDiv.innerHTML='';
    D3State.structItems.forEach(function(item,idx){
        var div=document.createElement('div'); div.className='struct-item';
        div.innerHTML='<span>📦 '+item.name+'</span><span style="font-size:10px;">('+item.pos.x+','+item.pos.y+','+item.pos.z+')</span>';
        div.onclick=function(){
            var newName=prompt('修改名称:',item.name);
            if(newName){ item.name=newName; D3State.saveStructsToStorage(); refreshStructList(); }
        };
        listDiv.appendChild(div);
    });
    D3State.structGroups.forEach(function(group,idx){
        var div=document.createElement('div'); div.className='struct-item struct-group';
        div.innerHTML='<span>📁 '+group.name+' ('+group.items.length+'个)</span><span style="font-size:10px;">组</span>';
        listDiv.appendChild(div);
    });
}
function syncMeterDisplay(){ document.getElementById('meterAD').value=D3State.meterAD.toFixed(2); document.getElementById('meterAE').value=D3State.meterAE.toFixed(2); }
function updateWorldParams(){
    var vab=parseInt(document.getElementById('voxelAB').value), vad=parseInt(document.getElementById('voxelAD').value), vae=parseInt(document.getElementById('voxelAE').value), mab=parseFloat(document.getElementById('meterAB').value);
    if(isNaN(vab)||vab<2) vab=2; if(vab>50) vab=50; if(isNaN(vad)||vad<2) vad=2; if(vad>50) vad=50; if(isNaN(vae)||vae<2) vae=2; if(vae>50) vae=50; if(isNaN(mab)||mab<=0) mab=3.0;
    document.getElementById('voxelAB').value=vab; document.getElementById('voxelAD').value=vad; document.getElementById('voxelAE').value=vae; document.getElementById('meterAB').value=mab;
    var oldVab=D3State.voxelCountAB,oldVad=D3State.voxelCountAD,oldVae=D3State.voxelCountAE;
    D3State.voxelCountAB=vab; D3State.voxelCountAD=vad; D3State.voxelCountAE=vae; D3State.meterAB=mab;
    D3State.recalcMeters(); syncMeterDisplay();
    if(oldVab!==vab||oldVad!==vad||oldVae!==vae) rebuildBoxAndGrid();
    refreshSceneVisuals();
}
