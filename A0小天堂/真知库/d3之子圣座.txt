    // ==================== 核心公式 ====================
    function voxelToGPS(voxel, origin) {
        let latStep = origin.meterPerVoxel / 111320.0;
        let lngStep = origin.meterPerVoxel / (111320.0 * Math.cos(origin.lat * Math.PI / 180.0));
        let lat = origin.lat + voxel.z * latStep;
        let lng = origin.lng + voxel.x * lngStep;
        let alt = origin.alt + voxel.y * origin.meterPerVoxel;
        return { lat, lng, alt };
    }

    function gpsToVoxel(gps, origin) {
        let latStep = origin.meterPerVoxel / 111320.0;
        let lngStep = origin.meterPerVoxel / (111320.0 * Math.cos(origin.lat * Math.PI / 180.0));
        let z = Math.round((gps.lat - origin.lat) / latStep);
        let x = Math.round((gps.lng - origin.lng) / lngStep);
        let y = Math.round((gps.alt - origin.alt) / origin.meterPerVoxel);
        return { x, y, z };
    }

    // 解析体素坐标文本
    function parseVoxelPoints(text) {
        let lines = text.split(/\r?\n/);
        let points = [];
        for (let line of lines) {
            let match = line.match(/(QD|D\d+|ZD)\((-?\d+),(-?\d+),(-?\d+)\)/);
            if (match) {
                points.push({ type: match[1], x: parseInt(match[2]), y: parseInt(match[3]), z: parseInt(match[4]) });
            }
        }
        return points;
    }

    // 从体素点生成表格数据
    function generateTableFromVoxel(points, origin) {
        if (!points.length) return { headers: defaultHeaders, data: [] };
        let data = [];
        for (let p of points) {
            let gps = voxelToGPS(p, origin);
            let row = {};
            for (let h of defaultHeaders) row[h] = getDefaultValue(h);
            row["latitude"] = gps.lat;
            row["longitude"] = gps.lng;
            row["altitude(m)"] = gps.alt;
            data.push(row);
        }
        return { headers: defaultHeaders, data };
    }

    function getDefaultValue(col) {
        if (col === "speed(m/s)") return 5.0;
        if (col === "altitudemode") return 0;
        if (col === "gimbalmode") return 0;
        if (col === "gimbalpitchangle") return 0;
        if (col === "heading(deg)") return 0;
        if (col.startsWith("actiontype")) return -1;
        if (col.startsWith("actionparam")) return 0;
        if (col === "curvesize(m)" || col === "rotationdir") return 0;
        if (col === "poi_latitude" || col === "poi_longitude" || col === "poi_altitude(m)" || col === "poi_altitudemode") return 0;
        if (col === "photo_timeinterval" || col === "photo_distinterval") return -1.0;
        return 0;
    }

    // 解析CSV文本
    function parseCSV(text) {
        let lines = text.split(/\r?\n/).filter(l => l.trim());
        if (lines.length < 2) return null;
        let headers = lines[0].split(',').map(h => h.trim());
        let data = [];
        for (let i = 1; i < lines.length; i++) {
            let vals = lines[i].split(',').map(v => v.trim());
            let row = {};
            for (let j = 0; j < headers.length && j < vals.length; j++) {
                let val = parseFloat(vals[j]);
                row[headers[j]] = isNaN(val) ? vals[j] : val;
            }
            data.push(row);
        }
        return { headers, data };
    }

    // 渲染表格
    function renderTable() {
        let wrapper = document.getElementById("tableWrapper");
        if (!currentHeaders.length || !currentData.length) {
            wrapper.innerHTML = '<div style="padding:20px; text-align:center; color:#999;">暂无数据，请先解析</div>';

        }
        let html = '<table><thead><tr>';
        for (let h of currentHeaders) {
            let chn = columnMapping[h] || h;
            html += `<th>${chn}</th>`;
        }
        html += '</tr></thead><tbody>';
        for (let i = 0; i < currentData.length; i++) {

    // 规则引擎
    function applyRuleToColumn(colName, ruleText) {
        if (!currentData.length) { showStatus("❌ 无数据"); return false; }
        let colIdx = currentHeaders.indexOf(colName);
        if (colIdx === -1) { showStatus(`❌ 未找到列 ${colName}`); return false; }

        let total = currentData.length;
        let rule = ruleText.trim();

        // 全局值
        let globalMatch = rule.match(/^(\d+(?:\.\d+)?)$/);
        if (globalMatch) {
            let val = parseFloat(globalMatch[1]);
            for (let i = 0; i < total; i++) currentData[i][colName] = val;
            showStatus(`✅ 已应用全局值 ${val} 到 ${colName}`);
            renderTable();
            return true;
        }

        // 分段规则
        if (rule.includes('=') && (rule.includes('QD') || rule.includes('D') || rule.includes('ZD'))) {
            let segments = rule.split(',');
            for (let seg of segments) {
                let [range, valStr] = seg.split('=');
                if (!range || !valStr) continue;
                let val = parseFloat(valStr.trim());
                if (isNaN(val)) continue;
                let startIdx = 0, endIdx = total - 1;
                let rangePart = range.trim();
                if (rangePart === 'QD') { startIdx = 0; endIdx = 0; }
                else if (rangePart === 'ZD') { startIdx = total - 1; endIdx = total - 1; }
                else if (rangePart.includes('-')) {
                    let [s, e] = rangePart.split('-');
                    if (s === 'QD') startIdx = 0;
                    else if (s.startsWith('D')) startIdx = parseInt(s.substring(1)) - 1;
                    else startIdx = parseInt(s) - 1;
                    if (e === 'ZD') endIdx = total - 1;
                    else if (e.startsWith('D')) endIdx = parseInt(e.substring(1)) - 1;
                    else endIdx = parseInt(e) - 1;
                } else if (rangePart.startsWith('D')) {
                    let idx = parseInt(rangePart.substring(1)) - 1;
                    startIdx = endIdx = idx;
                } else {
                    let idx = parseInt(rangePart) - 1;
                    startIdx = endIdx = idx;
                }
                startIdx = Math.max(0, Math.min(startIdx, total - 1));
                endIdx = Math.max(0, Math.min(endIdx, total - 1));
                for (let i = startIdx; i <= endIdx; i++) currentData[i][colName] = val;
            }
            showStatus(`✅ 已应用分段规则到 ${colName}`);
            renderTable();
            return true;
        }

        // 奇偶
        let oddEven = rule.match(/奇偶\((\d+(?:\.\d+)?),(\d+(?:\.\d+)?)\)/);
        if (oddEven) {
            let oddVal = parseFloat(oddEven[1]), evenVal = parseFloat(oddEven[2]);
            for (let i = 0; i < total; i++) {
                currentData[i][colName] = (i % 2 === 0) ? evenVal : oddVal;
            }
            showStatus(`✅ 已应用奇偶规则 (奇:${oddVal}, 偶:${evenVal})`);
            renderTable();
            return true;
        }

        // 质数
        let prime = rule.match(/质数\((\d+(?:\.\d+)?)\)/);
        if (prime) {
            let primeVal = parseFloat(prime[1]);
            function isPrime(n) { if (n<=1) return false; if (n<=3) return true; if (n%2===0||n%3===0) return false; for(let i=5;i*i<=n;i+=6) if(n%i===0||n%(i+2)===0) return false; return true; }
            for (let i = 0; i < total; i++) {
                if (isPrime(i+1)) currentData[i][colName] = primeVal;
            }
            showStatus(`✅ 已应用质数规则 (质数点:${primeVal})`);
            renderTable();
            return true;
        }

        // 正弦波
        let sin = rule.match(/正弦波\((\d+(?:\.\d+)?),(\d+(?:\.\d+)?),(\d+(?:\.\d+)?)\)/);
        if (sin) {
            let minVal = parseFloat(sin[1]), maxVal = parseFloat(sin[2]), period = parseFloat(sin[3]);
            if (period <= 0) period = total;
            for (let i = 0; i < total; i++) {
                let t = (i / period) * 2 * Math.PI;
                let sinVal = minVal + (maxVal - minVal) * (Math.sin(t) + 1) / 2;
                currentData[i][colName] = sinVal;
            }
            showStatus(`✅ 已应用正弦波规则 (${minVal}~${maxVal}, 周期${period})`);
            renderTable();
            return true;
        }

        // 斐波那契
        let fib = rule.match(/斐波那契\((\d+(?:\.\d+)?),(\d+(?:\.\d+)?)\)/);
        if (fib) {
            let f1 = parseFloat(fib[1]), f2 = parseFloat(fib[2]);
            let fibVals = [f1, f2];
            for (let i = 2; i < total; i++) fibVals.push(fibVals[i-1] + fibVals[i-2]);
            let maxFib = Math.max(...fibVals);
            for (let i = 0; i < total; i++) {
                let norm = f1 + (fibVals[i] / maxFib) * (f2 - f1);
                currentData[i][colName] = norm;
            }
            showStatus(`✅ 已应用斐波那契规则 (${f1}~${f2})`);
            renderTable();
            return true;
        }

        showStatus(`❌ 无法解析规则: ${rule}`);
        return false;
    }

    // 逆向生成体素流
    function reverseToVoxel() {
        if (!currentData.length || !currentHeaders.includes("latitude")) {
            showStatus("❌ 表格缺少经纬度数据");
            return;
        }
        let origin = { lat: currentData[0]["latitude"], lng: currentData[0]["longitude"], alt: currentData[0]["altitude(m)"], meterPerVoxel: currentOrigin.meterPerVoxel };
        let points = [];
        for (let i = 0; i < currentData.length; i++) {
            let gps = { lat: currentData[i]["latitude"], lng: currentData[i]["longitude"], alt: currentData[i]["altitude(m)"] };
            let vox = gpsToVoxel(gps, origin);
            let type = (i === 0) ? "QD" : (i === currentData.length-1) ? "ZD" : `D${i+1}`;
            points.push({ type, x: vox.x, y: vox.y, z: vox.z });
        }
        let output = points.map(p => `${p.type}(${p.x},${p.y},${p.z})`).join('\n');
        document.getElementById("voxelStreamOutput").innerText = output;
        document.getElementById("reverseResult").style.display = "block";
        showStatus("✅ 逆向生成体素坐标流完成");
    }

    // 搜索
    function performSearch() {
        let keyword = document.getElementById("searchInput").value.trim().toLowerCase();
        if (!keyword) { clearHighlight(); return; }
        let cells = document.querySelectorAll('td.editable');
        searchResults = [];
        cells.forEach((cell, idx) => {
            let text = cell.innerText.toLowerCase();
            if (text.includes(keyword)) {
                searchResults.push(cell);
                cell.classList.add('search-highlight');
            } else {
                cell.classList.remove('search-highlight');
            }
        });
        if (searchResults.length) { searchIndex = 0; searchResults[0].scrollIntoView({ behavior: 'smooth', block: 'center' }); }
        showStatus(`找到 ${searchResults.length} 个匹配项`);
    }
    function clearHighlight() {
        document.querySelectorAll('td.editable').forEach(cell => cell.classList.remove('search-highlight'));
        searchResults = []; searchIndex = -1;
    }
    function navigateSearch(delta) {
        if (!searchResults.length) return;
        searchIndex = (searchIndex + delta + searchResults.length) % searchResults.length;
        searchResults[searchIndex].scrollIntoView({ behavior: 'smooth', block: 'center' });
    }
