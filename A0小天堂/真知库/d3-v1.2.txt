        const D3State = {
            worldSize: { ad: 4, ab: 3, ae: 2.5 },
            voxelCount: 22,
            voxelPhysicalSize: { x: 0, y: 0, z: 0 },
            pathPoints: [],
            anchorMeshes: [],
            lockTranslate: false,
            lockRotate: false,
            lockScale: false,
            opacityFace: 0.35,
            opacityGrid: 0.5,
            scene: null,
            externalRenderer: null,
            internalRenderer: null,
            externalCamera: null,
            internalCamera: null,
            externalControls: null,

            redMidpoint: null,

            init() { this.calcVoxelSize(); },
            calcVoxelSize() {
                this.voxelPhysicalSize.x = this.worldSize.ab / this.voxelCount;
                this.voxelPhysicalSize.z = this.worldSize.ad / this.voxelCount;
                this.voxelPhysicalSize.y = this.worldSize.ae / this.voxelCount;
            },
            snapToCorner(x, y, z) {
                const sx = Math.round(x / this.voxelPhysicalSize.x) * this.voxelPhysicalSize.x;
                const sy = Math.round(y / this.voxelPhysicalSize.y) * this.voxelPhysicalSize.y;
                const sz = Math.round(z / this.voxelPhysicalSize.z) * this.voxelPhysicalSize.z;
                return { x: parseFloat(sx.toFixed(4)), y: parseFloat(sy.toFixed(4)), z: parseFloat(sz.toFixed(
                        4)) };
            }
        };
        D3State.init();

        /* ========== 模块6：文本输入与路径管理 ========== */
        const coordInput = document.getElementById('coordInput');
        const outputArea = document.getElementById('outputArea');
        const statusBar = document.getElementById('statusBar');

        function parseCoordText(text) {
            const points = [];
            const regex = /(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g;
            let match;
            while ((match = regex.exec(text)) !== null) {
                points.push({
                    type: match[1],
                    x: parseFloat(match[2]),
                    y: parseFloat(match[3]),
                    z: parseFloat(match[4])
                });
            }
            if (points.length === 0) return [];
            const qd = points.filter(p => p.type === 'QD');
            const zd = points.filter(p => p.type === 'ZD');
            let dn = points.filter(p => p.type.startsWith('D') && p.type !== 'QD' && p.type !== 'ZD');
            dn.sort((a, b) => (parseInt(a.type.substring(1)) - parseInt(b.type.substring(1))));
            const ordered = [...qd, ...dn, ...zd];
            return ordered.map(p => {
                const s = D3State.snapToCorner(p.x, p.y, p.z);
                return { type: p.type, x: s.x, y: s.y, z: s.z };
            });
        }

        function generatePackage(points) {
            if (points.length < 2) return null;
            let out = '';
            if (points.length === 2) out += '路径类型：直线路径\n';
            else out += '路径类型：自由路径\n';
            out += `总段数：${points.length - 1}\n\n路径坐标清单：\n`;
            out += points.map(p => `${p.type}(${p.x},${p.y},${p.z})`).join(' → ') + '\n\n';
            for (let i = 0; i < points.length - 1; i++) {
                const a = points[i],
                    b = points[i + 1];
                out +=
                    `段${i+1}: legX=${Math.abs(b.x-a.x).toFixed(2)}m, legZ=${Math.abs(b.z-a.z).toFixed(2)}m, legY=${Math.abs(b.y-a.y).toFixed(2)}m\n`;
            }
            return out;
        }

        function updateAnchorsInScene() {
            if (!D3State.scene) return;
            D3State.anchorMeshes.forEach(m => D3State.scene.remove(m));
            D3State.anchorMeshes = [];
            D3State.pathPoints.forEach(p => {
                const geo = new THREE.BoxGeometry(
                    D3State.voxelPhysicalSize.x * 0.8,
                    D3State.voxelPhysicalSize.y * 0.8,
                    D3State.voxelPhysicalSize.z * 0.8
                );
                let color;
                if (p.type === 'QD') color = 0x00ff00;
                else if (p.type === 'ZD') color = 0x0088ff;
                else color = 0xffffff;
                const mat = new THREE.MeshBasicMaterial({ color });
                const mesh = new THREE.Mesh(geo, mat);
                mesh.position.set(p.x, p.y, p.z);
                D3State.scene.add(mesh);
                D3State.anchorMeshes.push(mesh);
            });
        }

        function updatePathLineInScene() {
            if (!D3State.scene) return;
            if (D3State.pathLine) D3State.scene.remove(D3State.pathLine);
            if (D3State.arrowMesh) D3State.scene.remove(D3State.arrowMesh);
            if (D3State.pathPoints.length < 2) return;
            const pts = D3State.pathPoints.map(p => new THREE.Vector3(p.x, p.y, p.z));
            const geo = new THREE.BufferGeometry().setFromPoints(pts);
            const mat = new THREE.LineBasicMaterial({ color: 0xff3333, linewidth: 2 });
            D3State.pathLine = new THREE.Line(geo, mat);
            D3State.scene.add(D3State.pathLine);
            const last = pts[pts.length - 1];
            const prev = pts[pts.length - 2] || pts[0];
            const dir = new THREE.Vector3().subVectors(last, prev).normalize();
            if (dir.length() < 0.001) dir.set(0, 0, 1);
            const arrowGeo = new THREE.ConeGeometry(0.15, 0.4, 8, 8);
            const arrowMat = new THREE.MeshBasicMaterial({ color: 0xff3333 });
            D3State.arrowMesh = new THREE.Mesh(arrowGeo, arrowMat);
            D3State.arrowMesh.position.copy(last);
            D3State.arrowMesh.quaternion.setFromUnitVectors(new THREE.Vector3(0, 1, 0), dir);
            D3State.scene.add(D3State.arrowMesh);
        }

        function updateMidpointMarker() {
            if (!D3State.scene || !D3State.redMidpoint) return;
            // 【修复】吸附到最近的体素左上角角坐标
            const snapped = D3State.snapToCorner(0, 0, 0);
            D3State.redMidpoint.position.set(snapped.x, snapped.y, snapped.z);
            D3State.redMidpoint.visible = true;
        }

        function refreshSceneVisuals() {
            updateAnchorsInScene();
            updatePathLineInScene();
            updateMidpointMarker();
        }

                statusBar.innerHTML = '⚠️ 缺少起点(QD)或终点(ZD)'; return;
            }
            D3State.pathPoints = points;
            refreshSceneVisuals();
            statusBar.innerHTML =

        function stopRenderLoop(windowName) {
            if (windowName === 'external' && D3State.externalAnimId) {
                cancelAnimationFrame(D3State.externalAnimId);
                D3State.externalAnimId = null;
                activeRenderLoops.external = false;
            }
            if (windowName === 'internal' && D3State.internalAnimId) {
                cancelAnimationFrame(D3State.internalAnimId);
                D3State.internalAnimId = null;
                activeRenderLoops.internal = false;
            }
            if (windowName === 'manhattan' && D3State.manhattanAnimId) {
                cancelAnimationFrame(D3State.manhattanAnimId);
                D3State.manhattanAnimId = null;
                activeRenderLoops.manhattan = false;
            }
        }

        function startRenderLoop(windowName) {
            if (windowName === 'external' && !activeRenderLoops.external && D3State.externalRenderer && D3State
                .externalCamera) {
                activeRenderLoops.external = true;

                function render() {
                    if (!activeRenderLoops.external) return;
                    D3State.externalRenderer.render(D3State.scene, D3State.externalCamera);
                    D3State.externalAnimId = requestAnimationFrame(render);
                }
                render();
            }
            if (windowName === 'internal' && !activeRenderLoops.internal && D3State.internalRenderer && D3State
                .internalCamera) {
                activeRenderLoops.internal = true;

                function render() {
                    if (!activeRenderLoops.internal) return;
                    D3State.internalRenderer.render(D3State.scene, D3State.internalCamera);
                    D3State.internalAnimId = requestAnimationFrame(render);
                }
                render();
            }
            if (windowName === 'manhattan' && !activeRenderLoops.manhattan) {
                activeRenderLoops.manhattan = true;
                drawManhattanWindow();
                activeRenderLoops.manhattan = false;
            }
        }

        function drawManhattanWindow() {
            const canvas = document.getElementById('manhattanCanvas');
            if (!canvas) return;
            const container = document.getElementById('manhattanPanel');
            canvas.width = container.clientWidth;
            canvas.height = container.clientHeight;
            const ctx = canvas.getContext('2d');
            const w = canvas.width;
            const h = canvas.height;
            ctx.clearRect(0, 0, w, h);
            ctx.fillStyle = '#121420';
            ctx.fillRect(0, 0, w, h);
            const gridSize = 20;
            const cols = Math.floor(w / gridSize);
            const rows = Math.floor(h / gridSize);
            ctx.strokeStyle = '#333';
            ctx.lineWidth = 0.5;
            for (let i = 0; i <= cols; i++) { ctx.beginPath();
                ctx.moveTo(i * gridSize, 0);
                ctx.lineTo(i * gridSize, rows * gridSize);
                ctx.stroke(); }
            for (let i = 0; i <= rows; i++) { ctx.beginPath();
                ctx.moveTo(0, i * gridSize);
                ctx.lineTo(cols * gridSize, i * gridSize);
                ctx.stroke(); }
            if (D3State.pathPoints.length > 0) {
                ctx.strokeStyle = '#ff3333';
                ctx.lineWidth = 2;
                ctx.beginPath();
                const first = D3State.pathPoints[0];
                const scaleX = w / D3State.worldSize.ab;
                const scaleZ = h / D3State.worldSize.ad;
                ctx.moveTo(first.x * scaleX, first.z * scaleZ);
                for (let i = 1; i < D3State.pathPoints.length; i++) {
                    ctx.lineTo(D3State.pathPoints[i].x * scaleX, D3State.pathPoints[i].z * scaleZ);
                }
                ctx.stroke();
            }
            ctx.fillStyle = '#fff';
            ctx.font = '12px monospace';
            ctx.fillText('二维曼哈顿俯视图', 10, 20);
        }

        function rebuildBoxAndGrid() {
            if (!D3State.scene) return;
            if (D3State.boxMesh) D3State.scene.remove(D3State.boxMesh);
            if (D3State.gridLines) D3State.scene.remove(D3State.gridLines);
            if (D3State.redMidpoint) D3State.scene.remove(D3State.redMidpoint);
            const { ab, ad, ae } = D3State.worldSize;
            const n = D3State.voxelCount;
            const boxGeo = new THREE.BoxGeometry(ab, ae, ad);
            const boxMat = new THREE.MeshBasicMaterial({
                color: 0x4682c8,
                transparent: true,
                opacity: D3State.opacityFace,
                side: THREE.DoubleSide
            });
            D3State.boxMesh = new THREE.Mesh(boxGeo, boxMat);
            D3State.scene.add(D3State.boxMesh);
            const gridGroup = new THREE.Group();
            const vs = D3State.voxelPhysicalSize;
            for (let ix = 0; ix <= n; ix++) {
                const x = -ab / 2 + ix * vs.x;
                const geo = new THREE.BufferGeometry().setFromPoints([
                    new THREE.Vector3(x, -ae / 2, -ad / 2), new THREE.Vector3(x, ae / 2, -ad / 2),
                    new THREE.Vector3(x, ae / 2, ad / 2), new THREE.Vector3(x, -ae / 2, ad / 2),
                    new THREE.Vector3(x, -ae / 2, -ad / 2)
                ]);
                gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color: 0x000000, transparent: true,
                    opacity: D3State.opacityGrid })));
            }
            for (let iz = 0; iz <= n; iz++) {
                const z = -ad / 2 + iz * vs.z;
                const geo = new THREE.BufferGeometry().setFromPoints([
                    new THREE.Vector3(-ab / 2, -ae / 2, z), new THREE.Vector3(ab / 2, -ae / 2, z),
                    new THREE.Vector3(ab / 2, ae / 2, z), new THREE.Vector3(-ab / 2, ae / 2, z),
                    new THREE.Vector3(-ab / 2, -ae / 2, z)
                ]);
                gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color: 0x000000, transparent: true,
                    opacity: D3State.opacityGrid })));
            }
            for (let iy = 0; iy <= n; iy++) {
                const y = -ae / 2 + iy * vs.y;
                const geo = new THREE.BufferGeometry().setFromPoints([
                    new THREE.Vector3(-ab / 2, y, -ad / 2), new THREE.Vector3(ab / 2, y, -ad / 2),
                    new THREE.Vector3(ab / 2, y, ad / 2), new THREE.Vector3(-ab / 2, y, ad / 2),
                    new THREE.Vector3(-ab / 2, y, -ad / 2)
                ]);
                gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color: 0x000000, transparent: true,
                    opacity: D3State.opacityGrid })));
            }
            D3State.gridLines = gridGroup;
            D3State.scene.add(D3State.gridLines);
            const midGeo = new THREE.BoxGeometry(vs.x * 0.6, vs.y * 0.6, vs.z * 0.6);
            const midMat = new THREE.MeshBasicMaterial({ color: 0xff0000 });
            D3State.redMidpoint = new THREE.Mesh(midGeo, midMat);
            D3State.scene.add(D3State.redMidpoint);
            updateMidpointMarker();
        }

        function createOrbitControls(camera, domElement) {
            let isDown = false,
                isRightDrag = false;
            let lastX = 0,
                lastY = 0;
            const target = new THREE.Vector3(0, 0, 0);
            let spherical = new THREE.Spherical();
            const offset = new THREE.Vector3().copy(camera.position).sub(target);
            spherical.setFromVector3(offset);
            domElement.addEventListener('pointerdown', (e) => {
                if (e.button === 2 || e.button === 1) { isRightDrag = true; } else { isRightDrag =
                    false; }
                isDown = true;
                lastX = e.clientX;
                lastY = e.clientY;
                e.preventDefault();
            });
            window.addEventListener('pointermove', (e) => {
                if (!isDown) return;
                const dx = e.clientX - lastX;
                const dy = e.clientY - lastY;
                lastX = e.clientX;
                lastY = e.clientY;
                if (isRightDrag) {
                    if (D3State.lockTranslate) return;
                    const panSpeed = 0.01;
                    const right = new THREE.Vector3();
                    const up = new THREE.Vector3();
                    camera.getWorldDirection(new THREE.Vector3());
                    right.crossVectors(camera.up, new THREE.Vector3().copy(camera.position).normalize())
                        .normalize();
                    up.copy(camera.up).normalize();
                    target.add(right.multiplyScalar(-dx * panSpeed));
                    target.add(up.multiplyScalar(dy * panSpeed));
                } else {
                    if (D3State.lockRotate) return;
                    spherical.theta -= dx * 0.005;
                    spherical.phi -= dy * 0.005;
                    spherical.phi = Math.max(0.1, Math.min(Math.PI - 0.1, spherical.phi));
                }
                updateCameraFromSpherical();
            });
            window.addEventListener('pointerup', () => { isDown = false;
                isRightDrag = false; });
            domElement.addEventListener('wheel', (e) => {
                if (D3State.lockScale) return;
                spherical.radius *= (e.deltaY > 0) ? 1.05 : 0.95;
                spherical.radius = Math.max(1, Math.min(20, spherical.radius));
                updateCameraFromSpherical();
                e.preventDefault();
            });
            domElement.addEventListener('contextmenu', (e) => e.preventDefault());

            function updateCameraFromSpherical() {
                offset.setFromSpherical(spherical);
                camera.position.copy(target).add(offset);
                camera.lookAt(target);
            }
            return {
                target,
                update: () => { spherical.setFromVector3(new THREE.Vector3().copy(camera.position).sub(
                        target)); }
            };
        }
