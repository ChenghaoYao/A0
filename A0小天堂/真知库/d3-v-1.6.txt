var D3State = {
    // 体素数量（三方向独立）
    voxelCountAB: 22, voxelCountAD: 22, voxelCountAE: 22,
    // 物理米数（仅用于指令包输出换算）
    meterAB: 3.0, meterAD: 4.0, meterAE: 2.5,
    // 体素物理边长（自动计算，仅用于输出换算和渲染缩放）
    voxelSizeAB: 0, voxelSizeAD: 0, voxelSizeAE: 0,
    // 路径数据（坐标全部为整数体素索引）
    pathPoints: [],
    anchorMeshes: [], manualPlaceOrder: [],
    lockTranslate: false, lockRotate: false, lockScale: false,
    opacityFace: 0.35, opacityGrid: 0.5,
    scene: null, externalRenderer: null, internalRenderer: null,
    externalCamera: null, internalCamera: null, externalControls: null,
    boxMesh: null, gridLines: null, redMidpoint: null,
    pathLine: null, arrowMesh: null,
    externalAnimId: null, internalAnimId: null, manhattanAnimId: null,
    internalMode: 'observe',
    internalMoveForward: false, internalMoveBack: false, internalMoveLeft: false, internalMoveRight: false,
    internalMoveUp: false, internalMoveDown: false,
    internalMouseDown: false, internalLastMouseX: 0, internalLastMouseY: 0,
    internalPitch: 0, internalYaw: 0, internalReady: false,
    endpointClickCount: {},
    bleDevice: null, bleCharacteristic: null, wifiSocket: null,

    init: function() { this.calcVoxelSize(); },
    calcVoxelSize: function() {
        // 体素渲染尺寸 = 物理米数 / 体素数量（用于Three.js渲染缩放）
        this.voxelSizeAB = this.meterAB / this.voxelCountAB;
        this.voxelSizeAD = this.meterAD / this.voxelCountAD;
        this.voxelSizeAE = this.meterAE / this.voxelCountAE;
    },
    // 体素坐标吸附：整数取整（V1.6核心改动）
    snapToCorner: function(x, y, z) {
        return {
            x: Math.round(x),
            y: Math.round(y),
            z: Math.round(z)
        };
    },
    // 获取体素在Three.js场景中的渲染中心位置
    getVoxelRenderCenter: function(voxelCoord) {
        // 长方体几何中心在原点，体素坐标需映射到渲染空间
        var rx = (voxelCoord.x - this.voxelCountAB/2 + 0.5) * this.voxelSizeAB;
        var ry = (voxelCoord.y - this.voxelCountAE/2 + 0.5) * this.voxelSizeAE;
        var rz = (voxelCoord.z - this.voxelCountAD/2 + 0.5) * this.voxelSizeAD;
        return { x: rx, y: ry, z: rz };
    },
    // 从Three.js世界坐标反算体素坐标
    worldToVoxel: function(wx, wy, wz) {
        var vx = Math.round(wx / this.voxelSizeAB + this.voxelCountAB/2 - 0.5);
        var vy = Math.round(wy / this.voxelSizeAE + this.voxelCountAE/2 - 0.5);
        var vz = Math.round(wz / this.voxelSizeAD + this.voxelCountAD/2 - 0.5);
        return this.snapToCorner(vx, vy, vz);
    },
    isSnapEqual: function(a, b) {
        return a.x === b.x && a.y === b.y && a.z === b.z;
    },
    syncPathToText: function() {
        var lines = [];
        for (var i = 0; i < this.pathPoints.length; i++) {
            var p = this.pathPoints[i];
            lines.push(p.type + '(' + p.x + ',' + p.y + ',' + p.z + ')');
        }
        var el = document.getElementById('coordInput');
        if (el) el.value = lines.join('\n');
    }
};

/* ========== 路径解析 ========== */
function parseCoordText(text) {
    var tpMatch = text.match(/TP\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/);
    if (tpMatch) {
        var tx = parseInt(tpMatch[1]), ty = parseInt(tpMatch[2]), tz = parseInt(tpMatch[3]);
        if (D3State.internalCamera) {
            var rc = D3State.getVoxelRenderCenter({x:tx,y:ty,z:tz});
            D3State.internalCamera.position.set(rc.x, rc.y, rc.z + 2);
            D3State.internalCamera.lookAt(rc.x, rc.y, rc.z);
            if (statusBar) statusBar.innerHTML = '已传送到体素(' + tx + ', ' + ty + ', ' + tz + ')';
        }
        return [];
    }
    var points = [];
    var regex = /(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g;
    var match;
    while ((match = regex.exec(text)) !== null) {
        points.push({ type: match[1], x: parseInt(match[2]), y: parseInt(match[3]), z: parseInt(match[4]) });
    }
    if (points.length === 0) return [];
    var qd = [], zd = [], dn = [];
    for (var i = 0; i < points.length; i++) {
        if (points[i].type === 'QD') qd.push(points[i]);
        else if (points[i].type === 'ZD') zd.push(points[i]);
        else if (points[i].type.indexOf('D') === 0) dn.push(points[i]);
    }
    dn.sort(function(a, b) { return parseInt(a.type.substring(1)) - parseInt(b.type.substring(1)); });
    return qd.concat(dn).concat(zd);
}

function generatePackage(points) {
    if (points.length < 2) return null;
    var out = points.length === 2 ? '路径类型：直线路径\n' : '路径类型：自由路径\n';
    out += '总段数：' + (points.length - 1) + '\n\n路径坐标清单（体素）：\n';
    var names = [];
    for (var i = 0; i < points.length; i++) {
        names.push(points[i].type + '(' + points[i].x + ',' + points[i].y + ',' + points[i].z + ')');
    }
    out += names.join(' → ') + '\n\n物理距离清单：\n';
    for (var j = 0; j < points.length - 1; j++) {
        var a = points[j], b = points[j+1];
        var dx = Math.abs(b.x - a.x) * D3State.voxelSizeAB;
        var dz = Math.abs(b.z - a.z) * D3State.voxelSizeAD;
        var dy = Math.abs(b.y - a.y) * D3State.voxelSizeAE;
        out += '段' + (j+1) + ': legX=' + dx.toFixed(2) + 'm, legZ=' + dz.toFixed(2) + 'm, legY=' + dy.toFixed(2) + 'm\n';
    }
    return out;
}

/* ========== 场景更新 ========== */
function rebuildAllAnchorsFromPath() {
    if (!D3State.scene) return;
    for (var i = 0; i < D3State.anchorMeshes.length; i++) { D3State.scene.remove(D3State.anchorMeshes[i]); }
    D3State.anchorMeshes = [];
    D3State.manualPlaceOrder = [];
    for (var j = 0; j < D3State.pathPoints.length; j++) {
        var p = D3State.pathPoints[j];
        var rc = D3State.getVoxelRenderCenter(p);
        var geo = new THREE.BoxGeometry(D3State.voxelSizeAB*0.95, D3State.voxelSizeAE*0.95, D3State.voxelSizeAD*0.95);
        var color;
        if (p.type === 'QD') color = 0x00ff00;
        else if (p.type === 'ZD') color = 0x0088ff;
        else color = 0xffffff;
        var mesh = new THREE.Mesh(geo, new THREE.MeshBasicMaterial({ color: color }));
        mesh.position.set(rc.x, rc.y, rc.z);
        mesh.userData = { type: p.type, vx: p.x, vy: p.y, vz: p.z };
        D3State.scene.add(mesh);
        D3State.anchorMeshes.push(mesh);
        if (p.type === 'QD' || p.type.indexOf('D') === 0) { D3State.manualPlaceOrder.push(mesh); }
    }
}

function updatePathLineInScene() {
    if (!D3State.scene) return;
    if (D3State.pathLine) { D3State.scene.remove(D3State.pathLine); D3State.pathLine = null; }
    if (D3State.arrowMesh) { D3State.scene.remove(D3State.arrowMesh); D3State.arrowMesh = null; }
    if (D3State.pathPoints.length < 2) return;
    var pts = [];
    for (var i = 0; i < D3State.pathPoints.length; i++) {
        var rc = D3State.getVoxelRenderCenter(D3State.pathPoints[i]);
        pts.push(new THREE.Vector3(rc.x, rc.y, rc.z));
    }
    var geo = new THREE.BufferGeometry().setFromPoints(pts);
    D3State.pathLine = new THREE.Line(geo, new THREE.LineBasicMaterial({ color: 0xff3333, linewidth: 2 }));
    D3State.scene.add(D3State.pathLine);
    var last = pts[pts.length-1], prev = pts[pts.length-2] || pts[0];
    var dir = new THREE.Vector3().subVectors(last, prev).normalize();
    if (dir.length() < 0.001) dir.set(0,0,1);
    var arrowGeo = new THREE.ConeGeometry(0.2, 0.5, 8, 8);
    var arrowMesh = new THREE.Mesh(arrowGeo, new THREE.MeshBasicMaterial({ color: 0xff3333 }));
    arrowMesh.position.copy(last);
    arrowMesh.quaternion.setFromUnitVectors(new THREE.Vector3(0,1,0), dir);
    D3State.arrowMesh = arrowMesh;
    D3State.scene.add(arrowMesh);
}

function updateMidpointMarker() {
    if (!D3State.scene || !D3State.redMidpoint) return;
    var rc = D3State.getVoxelRenderCenter({x:0,y:0,z:0});
    D3State.redMidpoint.position.set(rc.x, rc.y, rc.z);
    D3State.redMidpoint.visible = true;
}

function refreshSceneVisuals() {
    rebuildAllAnchorsFromPath();
    updatePathLineInScene();
    updateMidpointMarker();
}

function collapseAfterIndex(idx) {
    if (idx < 0) idx = -1;
    if (idx >= D3State.pathPoints.length) return;
    D3State.pathPoints = D3State.pathPoints.slice(0, idx + 1);
    D3State.syncPathToText();
    refreshSceneVisuals();
}

function addManualPoint(voxelCoord) {
    for (var i = 0; i < D3State.pathPoints.length; i++) {
        if (D3State.isSnapEqual(D3State.pathPoints[i], voxelCoord)) {
            if (statusBar) statusBar.innerHTML = '该位置已存在路径点';
            return;
        }
    }
    var type;
    if (D3State.pathPoints.length === 0) { type = 'QD'; }
    else {
        var dnCount = 0;
        for (var j = 0; j < D3State.pathPoints.length; j++) {
            if (D3State.pathPoints[j].type.indexOf('D') === 0) dnCount++;
        }
        type = 'D' + (dnCount + 2);
    }
    D3State.pathPoints.push({ type: type, x: voxelCoord.x, y: voxelCoord.y, z: voxelCoord.z });
    D3State.syncPathToText();
    refreshSceneVisuals();
    if (statusBar) statusBar.innerHTML = '放置 ' + type + ' @ 体素(' + voxelCoord.x + ', ' + voxelCoord.y + ', ' + voxelCoord.z + ')';
}

function removePointByVoxel(voxelCoord) {
    var idx = -1;
    for (var i = 0; i < D3State.pathPoints.length; i++) {
        if (D3State.isSnapEqual(D3State.pathPoints[i], voxelCoord)) { idx = i; break; }
    }
    if (idx < 0) { if (statusBar) statusBar.innerHTML = '未找到对应路径点'; return; }
    collapseAfterIndex(idx - 1);
    if (statusBar) statusBar.innerHTML = '拆除并崩塌后续路径';
}

function markAsEndpoint(voxelCoord) {
    var idx = -1;
    for (var i = 0; i < D3State.pathPoints.length; i++) {
        if (D3State.isSnapEqual(D3State.pathPoints[i], voxelCoord)) { idx = i; break; }
    }
    if (idx < 0) { if (statusBar) statusBar.innerHTML = '请先放置该方块'; return; }
    D3State.pathPoints[idx].type = 'ZD';
    collapseAfterIndex(idx);
    if (statusBar) statusBar.innerHTML = '终点已标记 @ 体素(' + voxelCoord.x + ', ' + voxelCoord.y + ', ' + voxelCoord.z + ')';
}

/* ========== Tab切换 ========== */
var activeRenderLoops = { external:false, internal:false, manhattan:false, video:false };
function stopRenderLoop(name) {
    if (name === 'external' && D3State.externalAnimId) { cancelAnimationFrame(D3State.externalAnimId); D3State.externalAnimId = null; activeRenderLoops.external = false; }
    if (name === 'internal' && D3State.internalAnimId) { cancelAnimationFrame(D3State.internalAnimId); D3State.internalAnimId = null; activeRenderLoops.internal = false; }
    if (name === 'manhattan' && D3State.manhattanAnimId) { cancelAnimationFrame(D3State.manhattanAnimId); D3State.manhattanAnimId = null; activeRenderLoops.manhattan = false; }
}
function startRenderLoop(name) {
    if (name === 'external' && !activeRenderLoops.external && D3State.externalRenderer && D3State.externalCamera) {
        activeRenderLoops.external = true;
        (function loop() { if(!activeRenderLoops.external) return; D3State.externalRenderer.render(D3State.scene, D3State.externalCamera); D3State.externalAnimId = requestAnimationFrame(loop); })();
    }
    if (name === 'internal' && !activeRenderLoops.internal && D3State.internalRenderer && D3State.internalCamera) {
        if (!D3State.internalReady) { if(statusBar) statusBar.innerHTML = '内部窗口正在初始化，请稍后再试...'; return; }
        activeRenderLoops.internal = true;
        (function loop() { if(!activeRenderLoops.internal) return; if(typeof updateInternalMovement === 'function') updateInternalMovement(); D3State.internalRenderer.render(D3State.scene, D3State.internalCamera); D3State.internalAnimId = requestAnimationFrame(loop); })();
    }
    if (name === 'manhattan' && !activeRenderLoops.manhattan) { activeRenderLoops.manhattan = true; if(typeof drawManhattanWindow === 'function') drawManhattanWindow(); activeRenderLoops.manhattan = false; }
}

/* ========== 二维曼哈顿 ========== */
function drawManhattanWindow() {
    var canvas = document.getElementById('manhattanCanvas');
    if (!canvas) return;
    var w = document.getElementById('windowContainer').clientWidth;
    var h = document.getElementById('windowContainer').clientHeight;
    canvas.width = w; canvas.height = h;
    var ctx = canvas.getContext('2d');
    ctx.clearRect(0,0,w,h); ctx.fillStyle = '#121420'; ctx.fillRect(0,0,w,h);
    var gs = 15, cols = Math.floor(w/gs), rows = Math.floor(h/gs);
    ctx.strokeStyle = '#333'; ctx.lineWidth = 0.5;
    for (var i=0;i<=cols;i++) { ctx.beginPath(); ctx.moveTo(i*gs,0); ctx.lineTo(i*gs,rows*gs); ctx.stroke(); }
    for (var j=0;j<=rows;j++) { ctx.beginPath(); ctx.moveTo(0,j*gs); ctx.lineTo(cols*gs,j*gs); ctx.stroke(); }
    if (D3State.pathPoints.length > 0) {
        ctx.strokeStyle = '#ff3333'; ctx.lineWidth = 2; ctx.beginPath();
        var sx = w/D3State.voxelCountAB, sz = h/D3State.voxelCountAD;
        ctx.moveTo(D3State.pathPoints[0].x*sx, D3State.pathPoints[0].z*sz);
        for (var k=1;k<D3State.pathPoints.length;k++) { ctx.lineTo(D3State.pathPoints[k].x*sx, D3State.pathPoints[k].z*sz); }
        ctx.stroke();
    }
    ctx.fillStyle = '#fff'; ctx.font = '12px monospace'; ctx.fillText('二维曼哈顿俯视图', 10, 20);
}

/* ========== 内部窗口移动 ========== */
function updateInternalMovement() {
    if (!D3State.internalCamera) return;
    var speed = 0.15;
    var forward = new THREE.Vector3(0,0,-1).applyQuaternion(D3State.internalCamera.quaternion);
    forward.y = 0; forward.normalize();
    var right = new THREE.Vector3(1,0,0).applyQuaternion(D3State.internalCamera.quaternion);
    right.y = 0; right.normalize();
    if (D3State.internalMoveForward) D3State.internalCamera.position.addScaledVector(forward, speed);
    if (D3State.internalMoveBack) D3State.internalCamera.position.addScaledVector(forward, -speed);
    if (D3State.internalMoveRight) D3State.internalCamera.position.addScaledVector(right, speed);
    if (D3State.internalMoveLeft) D3State.internalCamera.position.addScaledVector(right, -speed);
    if (D3State.internalMoveUp) D3State.internalCamera.position.y += speed;
    if (D3State.internalMoveDown) D3State.internalCamera.position.y -= speed;
    var coordsEl = document.getElementById('internalCoords');
    if (coordsEl) {
        var vc = D3State.worldToVoxel(D3State.internalCamera.position.x, D3State.internalCamera.position.y, D3State.internalCamera.position.z);
        coordsEl.textContent = '体素(' + vc.x + ', ' + vc.y + ', ' + vc.z + ')';
    }
}

/* ========== 内部窗口交互 ========== */
function setupInternalControls() {
    var intPanel = document.getElementById('internalPanel');
    if (!intPanel) return;
    window.addEventListener('keydown', function(e) {
        if (!document.getElementById('internalPanel').classList.contains('active')) return;
        switch(e.key.toLowerCase()) {
            case 'w': D3State.internalMoveForward = true; e.preventDefault(); break;
            case 's': D3State.internalMoveBack = true; e.preventDefault(); break;
            case 'a': D3State.internalMoveLeft = true; e.preventDefault(); break;
            case 'd': D3State.internalMoveRight = true; e.preventDefault(); break;
            case 'q': D3State.internalMoveDown = true; e.preventDefault(); break;
            case 'e': D3State.internalMoveUp = true; e.preventDefault(); break;
        }
    });
    window.addEventListener('keyup', function(e) {
        switch(e.key.toLowerCase()) {
            case 'w': D3State.internalMoveForward = false; e.preventDefault(); break;
            case 's': D3State.internalMoveBack = false; e.preventDefault(); break;
            case 'a': D3State.internalMoveLeft = false; e.preventDefault(); break;
            case 'd': D3State.internalMoveRight = false; e.preventDefault(); break;
            case 'q': D3State.internalMoveDown = false; e.preventDefault(); break;
            case 'e': D3State.internalMoveUp = false; e.preventDefault(); break;
        }
    });
    var modeIndicator = document.getElementById('internalMode');
    window.addEventListener('keydown', function(e) {
        if (!document.getElementById('internalPanel').classList.contains('active')) return;
        if (e.key === '1') { D3State.internalMode = 'place'; if(modeIndicator) modeIndicator.textContent = '放置模式'; e.preventDefault(); }
        else if (e.key === '0') { D3State.internalMode = 'remove'; if(modeIndicator) modeIndicator.textContent = '拆除模式'; e.preventDefault(); }
        else if (e.key === '8') { D3State.internalMode = 'endpoint'; if(modeIndicator) modeIndicator.textContent = '终点模式'; e.preventDefault(); }
        else if (e.key === 'Escape') { D3State.internalMode = 'observe'; if(modeIndicator) modeIndicator.textContent = '观察模式'; e.preventDefault(); }
    });
    intPanel.addEventListener('pointerdown', function(e) { D3State.internalMouseDown = true; D3State.internalLastMouseX = e.clientX; D3State.internalLastMouseY = e.clientY; intPanel.setPointerCapture(e.pointerId); e.preventDefault(); });
    window.addEventListener('pointermove', function(e) {
        if (!D3State.internalMouseDown) return;
        if (!document.getElementById('internalPanel').classList.contains('active')) return;
        var dx = e.clientX - D3State.internalLastMouseX, dy = e.clientY - D3State.internalLastMouseY;
        D3State.internalLastMouseX = e.clientX; D3State.internalLastMouseY = e.clientY;
        D3State.internalYaw -= dx * 0.002; D3State.internalPitch -= dy * 0.002;
        D3State.internalPitch = Math.max(-Math.PI/2, Math.min(Math.PI/2, D3State.internalPitch));
        var q = new THREE.Quaternion(); q.setFromEuler(new THREE.Euler(D3State.internalPitch, D3State.internalYaw, 0, 'YXZ'));
        D3State.internalCamera.quaternion.copy(q);
    });
    window.addEventListener('pointerup', function() { D3State.internalMouseDown = false; });
    intPanel.addEventListener('click', function(e) {
        if (D3State.internalMode === 'observe') return;
        if (!D3State.scene || !D3State.boxMesh) return;
        var raycaster = new THREE.Raycaster();
        var mouse = new THREE.Vector2();
        var rect = intPanel.getBoundingClientRect();
        mouse.x = ((e.clientX - rect.left) / rect.width) * 2 - 1;
        mouse.y = -((e.clientY - rect.top) / rect.height) * 2 + 1;
        raycaster.setFromCamera(mouse, D3State.internalCamera);
        var intersects = raycaster.intersectObjects([D3State.boxMesh], false);
        if (intersects.length > 0) {
            var point = intersects[0].point;
            var vc = D3State.worldToVoxel(point.x, point.y, point.z);
            if (D3State.internalMode === 'place') { addManualPoint(vc); }
            else if (D3State.internalMode === 'remove') { removePointByVoxel(vc); }
            else if (D3State.internalMode === 'endpoint') {
                var key = vc.x + ',' + vc.y + ',' + vc.z;
                D3State.endpointClickCount[key] = (D3State.endpointClickCount[key] || 0) + 1;
                if (D3State.endpointClickCount[key] >= 3) { markAsEndpoint(vc); D3State.endpointClickCount = {}; }
                else { if(statusBar) statusBar.innerHTML = '终点模式：已点击 ' + D3State.endpointClickCount[key] + ' 次，需3次确认'; }
            }
        }
    });
}

function rebuildBoxAndGrid() {
    if (!D3State.scene) return;
    if (D3State.boxMesh) D3State.scene.remove(D3State.boxMesh);
    if (D3State.gridLines) D3State.scene.remove(D3State.gridLines);
    if (D3State.redMidpoint) D3State.scene.remove(D3State.redMidpoint);
    var bw = D3State.voxelCountAB * D3State.voxelSizeAB;
    var bh = D3State.voxelCountAE * D3State.voxelSizeAE;
    var bd = D3State.voxelCountAD * D3State.voxelSizeAD;
    D3State.boxMesh = new THREE.Mesh(new THREE.BoxGeometry(bw, bh, bd), new THREE.MeshBasicMaterial({ color: 0x4682c8, transparent: true, opacity: D3State.opacityFace, side: THREE.DoubleSide }));
    D3State.scene.add(D3State.boxMesh);
    var gridGroup = new THREE.Group();
    for (var ix=0;ix<=D3State.voxelCountAB;ix++) {
        var x = -bw/2 + ix*D3State.voxelSizeAB;
        var geo = new THREE.BufferGeometry().setFromPoints([ new THREE.Vector3(x,-bh/2,-bd/2),new THREE.Vector3(x,bh/2,-bd/2),new THREE.Vector3(x,bh/2,bd/2),new THREE.Vector3(x,-bh/2,bd/2),new THREE.Vector3(x,-bh/2,-bd/2)]);
        gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color:0x000000, transparent:true, opacity:D3State.opacityGrid })));
    }
    for (var iz=0;iz<=D3State.voxelCountAD;iz++) {
        var z = -bd/2 + iz*D3State.voxelSizeAD;
        var geo = new THREE.BufferGeometry().setFromPoints([ new THREE.Vector3(-bw/2,-bh/2,z),new THREE.Vector3(bw/2,-bh/2,z),new THREE.Vector3(bw/2,bh/2,z),new THREE.Vector3(-bw/2,bh/2,z),new THREE.Vector3(-bw/2,-bh/2,z)]);
        gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color:0x000000, transparent:true, opacity:D3State.opacityGrid })));
    }
    for (var iy=0;iy<=D3State.voxelCountAE;iy++) {
        var y = -bh/2 + iy*D3State.voxelSizeAE;
        var geo = new THREE.BufferGeometry().setFromPoints([ new THREE.Vector3(-bw/2,y,-bd/2),new THREE.Vector3(bw/2,y,-bd/2),new THREE.Vector3(bw/2,y,bd/2),new THREE.Vector3(-bw/2,y,bd/2),new THREE.Vector3(-bw/2,y,-bd/2)]);
        gridGroup.add(new THREE.Line(geo, new THREE.LineBasicMaterial({ color:0x000000, transparent:true, opacity:D3State.opacityGrid })));
    }
    D3State.gridLines = gridGroup; D3State.scene.add(gridGroup);
    var vs = Math.min(D3State.voxelSizeAB, D3State.voxelSizeAD, D3State.voxelSizeAE);
    var midGeo = new THREE.BoxGeometry(vs, vs, vs);
    D3State.redMidpoint = new THREE.Mesh(midGeo, new THREE.MeshBasicMaterial({ color:0xff0000 }));
    D3State.scene.add(D3State.redMidpoint);
    updateMidpointMarker();
}

function createOrbitControls(camera, domElement) {
    var isDown=false, isRightDrag=false, lastX=0, lastY=0;
    var target = new THREE.Vector3(0,0,0);
    var spherical = new THREE.Spherical().setFromVector3(new THREE.Vector3().copy(camera.position).sub(target));
    domElement.addEventListener('pointerdown', function(e) { isRightDrag = (e.button===2||e.button===1); isDown=true; lastX=e.clientX; lastY=e.clientY; e.preventDefault(); });
    window.addEventListener('pointermove', function(e) {
        if (!isDown) return;
        var dx=e.clientX-lastX, dy=e.clientY-lastY; lastX=e.clientX; lastY=e.clientY;
        if (isRightDrag) {
            if (D3State.lockTranslate) return;
            var right = new THREE.Vector3().crossVectors(camera.up, new THREE.Vector3().copy(camera.position).normalize()).normalize();
            var up = camera.up.clone().normalize();
            target.add(right.multiplyScalar(-dx*0.01)).add(up.multiplyScalar(dy*0.01));
        } else {
            if (D3State.lockRotate) return;
            spherical.theta -= dx*0.005; spherical.phi -= dy*0.005;
            spherical.phi = Math.max(0.1, Math.min(Math.PI-0.1, spherical.phi));
        }
        var offset = new THREE.Vector3().setFromSpherical(spherical);
        camera.position.copy(target).add(offset); camera.lookAt(target);
    });
    window.addEventListener('pointerup', function() { isDown=false; isRightDrag=false; });
    domElement.addEventListener('wheel', function(e) {
        if (D3State.lockScale) return;
        spherical.radius *= (e.deltaY>0)?1.05:0.95;
        spherical.radius = Math.max(0.05, Math.min(50, spherical.radius));
        var offset = new THREE.Vector3().setFromSpherical(spherical);
        camera.position.copy(target).add(offset); camera.lookAt(target);
        e.preventDefault();
    });
    domElement.addEventListener('contextmenu', function(e) { e.preventDefault(); });
    return { target: target, update: function() { spherical.setFromVector3(new THREE.Vector3().copy(camera.position).sub(target)); } };
}

function updateWorldParams() {
    var vab = parseInt(document.getElementById('voxelAB').value);
    var vad = parseInt(document.getElementById('voxelAD').value);
    var vae = parseInt(document.getElementById('voxelAE').value);
    var mab = parseFloat(document.getElementById('meterAB').value);
    var mad = parseFloat(document.getElementById('meterAD').value);
    var mae = parseFloat(document.getElementById('meterAE').value);
    if (isNaN(vab) || vab < 2) vab = 2; if (vab > 33) vab = 33;
    if (isNaN(vad) || vad < 2) vad = 2; if (vad > 33) vad = 33;
    if (isNaN(vae) || vae < 2) vae = 2; if (vae > 33) vae = 33;
    if (isNaN(mab) || mab <= 0) mab = 3.0;
    if (isNaN(mad) || mad <= 0) mad = 4.0;
    if (isNaN(mae) || mae <= 0) mae = 2.5;
    document.getElementById('voxelAB').value = vab;
    document.getElementById('voxelAD').value = vad;
    document.getElementById('voxelAE').value = vae;
    document.getElementById('meterAB').value = mab;
    document.getElementById('meterAD').value = mad;
    document.getElementById('meterAE').value = mae;
    D3State.voxelCountAB = vab; D3State.voxelCountAD = vad; D3State.voxelCountAE = vae;
    D3State.meterAB = mab; D3State.meterAD = mad; D3State.meterAE = mae;
    D3State.calcVoxelSize();
    rebuildBoxAndGrid();
    refreshSceneVisuals();
}
