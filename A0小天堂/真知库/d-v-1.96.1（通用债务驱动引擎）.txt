<script>
// ========== 模块A：绝对坐标系 ==========
var AbsCoords = {
  voxelRenderSize: 1.0,
  snapToCorner: function(x,y,z){ return {x:Math.round(x),y:Math.round(y),z:Math.round(z)}; },
  getVoxelRenderCenter: function(vc){ return {x:vc.x*this.voxelRenderSize, y:vc.y*this.voxelRenderSize, z:vc.z*this.voxelRenderSize}; },
  worldToVoxel: function(wx,wy,wz){ return this.snapToCorner(Math.round(wx/this.voxelRenderSize),Math.round(wy/this.voxelRenderSize),Math.round(wz/this.voxelRenderSize)); },
  isSnapEqual: function(a,b){ return a.x===b.x && a.y===b.y && a.z===b.z; }
};

// ========== 模块B：固定坐标系 ==========
var FixedFrame = {
  voxelAB:22, voxelAD:22, voxelAE:22, meterAB:3.0, meterAD:3.0, meterAE:3.0,
  showCornerLabels:false, penetrateMode:false,
  hollowBoxMeshes:[], cornerLabelSprites:[], scene:null, redMidpoint:null,
  opacityFace:0.35, opacityGrid:0.5, gridLines:null,
  initScene:function(s){ this.scene=s; },
  recalcMeters:function(){ this.meterAD=this.meterAB*(this.voxelAD/this.voxelAB); this.meterAE=this.meterAB*(this.voxelAE/this.voxelAB); },
  getVoxelMinMax:function(){
    var hx=Math.floor(this.voxelAB/2), hy=Math.floor(this.voxelAE/2), hz=Math.floor(this.voxelAD/2);
    if(this.voxelAB%2===0){ return {xMin:-hx, xMax:hx-1, yMin:-hy, yMax:this.voxelAE%2===0?hy-1:hy, zMin:-hz, zMax:this.voxelAD%2===0?hz-1:hz}; }
    else{ return {xMin:-hx, xMax:hx, yMin:-hy, yMax:hy, zMin:-hz, zMax:hz}; }
  },
  rebuildHollowBox:function(){
    if(!this.scene) return;
    for(var i=0;i<this.hollowBoxMeshes.length;i++) this.scene.remove(this.hollowBoxMeshes[i]); this.hollowBoxMeshes=[];
    for(var j=0;j<this.cornerLabelSprites.length;j++) this.scene.remove(this.cornerLabelSprites[j]); this.cornerLabelSprites=[];
    if(this.gridLines){ this.scene.remove(this.gridLines); this.gridLines=null; }
    if(this.redMidpoint){ this.scene.remove(this.redMidpoint); this.redMidpoint=null; }
    var mm=this.getVoxelMinMax(), vs=AbsCoords.voxelRenderSize;
    for(var vx=mm.xMin; vx<=mm.xMax; vx++){ for(var vy=mm.yMin; vy<=mm.yMax; vy++){ for(var vz=mm.zMin; vz<=mm.zMax; vz++){
      if(!(vx===mm.xMin||vx===mm.xMax||vy===mm.yMin||vy===mm.yMax||vz===mm.zMin||vz===mm.zMax)) continue;
      var rc=AbsCoords.getVoxelRenderCenter({x:vx,y:vy,z:vz});
      var geo=new THREE.BoxGeometry(vs*0.95, vs*0.95, vs*0.95);
      var mat=new THREE.MeshBasicMaterial({color:0x3366aa, transparent:true, opacity:0.45});
      var mesh=new THREE.Mesh(geo,mat); mesh.position.set(rc.x,rc.y,rc.z);
      mesh.userData={type:'frameBlock', vx:vx, vy:vy, vz:vz};
      mesh.add(new THREE.LineSegments(new THREE.EdgesGeometry(geo), new THREE.LineBasicMaterial({color:0x000000})));
      this.scene.add(mesh); this.hollowBoxMeshes.push(mesh);
    }}}
    var midGeo=new THREE.BoxGeometry(vs,vs,vs); this.redMidpoint=new THREE.Mesh(midGeo, new THREE.MeshBasicMaterial({color:0xff0000}));
    this.redMidpoint.position.set(0,0,0); this.scene.add(this.redMidpoint);
    this.gridLines=new THREE.Group();
    for(var ix=mm.xMin; ix<=mm.xMax+1; ix++){ var x=ix*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(x,mm.yMin*vs,mm.zMin*vs), new THREE.Vector3(x,mm.yMax*vs+vs,mm.zMin*vs), new THREE.Vector3(x,mm.yMax*vs+vs,mm.zMax*vs+vs), new THREE.Vector3(x,mm.yMin*vs,mm.zMax*vs+vs), new THREE.Vector3(x,mm.yMin*vs,mm.zMin*vs)]); this.gridLines.add(new THREE.Line(geo, new THREE.LineBasicMaterial({color:0x000000, transparent:true, opacity:this.opacityGrid}))); }
    for(var iz=mm.zMin; iz<=mm.zMax+1; iz++){ var z=iz*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(mm.xMin*vs,mm.yMin*vs,z), new THREE.Vector3(mm.xMax*vs+vs,mm.yMin*vs,z), new THREE.Vector3(mm.xMax*vs+vs,mm.yMax*vs+vs,z), new THREE.Vector3(mm.xMin*vs,mm.yMax*vs+vs,z), new THREE.Vector3(mm.xMin*vs,mm.yMin*vs,z)]); this.gridLines.add(new THREE.Line(geo, new THREE.LineBasicMaterial({color:0x000000, transparent:true, opacity:this.opacityGrid}))); }
    for(var iy=mm.yMin; iy<=mm.yMax+1; iy++){ var y=iy*vs; var geo=new THREE.BufferGeometry().setFromPoints([new THREE.Vector3(mm.xMin*vs,y,mm.zMin*vs), new THREE.Vector3(mm.xMax*vs+vs,y,mm.zMin*vs), new THREE.Vector3(mm.xMax*vs+vs,y,mm.zMax*vs+vs), new THREE.Vector3(mm.xMin*vs,y,mm.zMax*vs+vs), new THREE.Vector3(mm.xMin*vs,y,mm.zMin*vs)]); this.gridLines.add(new THREE.Line(geo, new THREE.LineBasicMaterial({color:0x000000, transparent:true, opacity:this.opacityGrid}))); }
    this.scene.add(this.gridLines);
    this.updateCornerLabels();
  },
  updateCornerLabels:function(){
    for(var j=0;j<this.cornerLabelSprites.length;j++) this.scene.remove(this.cornerLabelSprites[j]); this.cornerLabelSprites=[];
    if(!this.showCornerLabels) return;

// ========== 辅助函数 ==========
function voxelToGPS(voxel, originLat, originLng, originAlt, meterPerVoxel) {
  var latStep = meterPerVoxel / 111320.0;
  var lngStep = meterPerVoxel / (111320.0 * Math.cos(originLat * Math.PI / 180.0));
  var lat = originLat + voxel.z * latStep;
  var lng = originLng + voxel.x * lngStep;
  var alt = originAlt + voxel.y * meterPerVoxel;
  return { lat: lat, lng: lng, alt: alt };
}

function generateKML(points, originLat, originLng, originAlt, meterPerVoxel, missionName) {
  if (points.length < 2) return null;
  var style = '<Style id="wp-marker-style">' + '<IconStyle><color>ffff0076</color>' + '<Icon><href>http://maps.google.com/mapfiles/kml/paddle/wht-blank.png</href></Icon>' + '</IconStyle></Style>' + '<Style id="wpline-style-normal">' + '<LineStyle><color>ff00ffff</color><width>3</width></LineStyle>' + '</Style>' + '<StyleMap id="wpline-style">' + '<Pair><key>normal</key><styleUrl>#wpline-style-normal</styleUrl></Pair>' + '</StyleMap>';
  var kml = '<?xml version="1.0" encoding="UTF-8"?>\n';
  kml += '<kml xmlns="http://www.opengis.net/kml/2.2" xmlns:gx="http://www.google.com/kml/ext/2.2">\n';
  kml += '<Document>\n';
  kml += ' <name>' + missionName + '</name>\n';
  kml += style + '\n';
  kml += ' <Folder>\n <name>Waypoints</name>\n';
  for (var i = 0; i < points.length; i++) {
    var p = points[i];
    var gps = voxelToGPS(p, originLat, originLng, originAlt, meterPerVoxel);
    var heading = 0;
    if (i < points.length - 1) { var nextP = points[i+1]; var dx = nextP.x - p.x; var dz = nextP.z - p.z; heading = Math.atan2(dz, dx) * 180 / Math.PI; }
    else if (points.length > 1) { var prevP = points[i-1]; var dx2 = p.x - prevP.x; var dz2 = p.z - prevP.z; heading = Math.atan2(dz2, dx2) * 180 / Math.PI; }
    kml += ' <Placemark>\n';
    kml += ' <name>航点 ' + (i+1) + '</name>\n';
    kml += ' <styleUrl>#wp-marker-style</styleUrl>\n';
    kml += ' <Point>\n';
    kml += ' <extrude>1</extrude>\n';
    kml += ' <altitudeMode>absolute</altitudeMode>\n';
    kml += ' <coordinates>' + gps.lng.toFixed(7) + ',' + gps.lat.toFixed(7) + ',' + gps.alt.toFixed(2) + '</coordinates>\n';
    kml += ' </Point>\n';
    kml += ' <Camera>\n';
    kml += ' <gx:horizFov>69.7</gx:horizFov>\n';
    kml += ' <heading>' + heading.toFixed(1) + '</heading>\n';
    kml += ' <tilt>0</tilt>\n';
    kml += ' </Camera>\n';
    kml += ' </Placemark>\n';
  }
  kml += ' </Folder>\n';
  kml += ' <Placemark>\n';
  kml += ' <name>Waypoint Path</name>\n';
  kml += ' <styleUrl>#wpline-style</styleUrl>\n';
  kml += ' <LineString>\n';
  kml += ' <extrude>1</extrude>\n';
  kml += ' <altitudeMode>absolute</altitudeMode>\n';
  kml += ' <coordinates>\n';
  for (var i = 0; i < points.length; i++) { var gps = voxelToGPS(points[i], originLat, originLng, originAlt, meterPerVoxel); kml += ' ' + gps.lng.toFixed(7) + ',' + gps.lat.toFixed(7) + ',' + gps.alt.toFixed(2) + '\n'; }
  kml += ' </coordinates>\n';
  kml += ' </LineString>\n';
  kml += ' </Placemark>\n';
  kml += '</Document>\n</kml>';
  return kml;
}

var DefaultParams = { speed: 5.0, gimbalPitch: 0, heading: 0, altOffset: 0, gimbalMode: 0, actionType: -1 };

function initWaypointParams(count) {
  waypointParams = [];
  for (var i = 0; i < count; i++) {
    waypointParams.push({ speed: DefaultParams.speed, gimbalPitch: DefaultParams.gimbalPitch, heading: DefaultParams.heading, altOffset: DefaultParams.altOffset, gimbalMode: DefaultParams.gimbalMode, actionType: DefaultParams.actionType });
  }
}

function isPrime(n) { if (n <= 1) return false; if (n <= 3) return true; if (n % 2 === 0 || n % 3 === 0) return false; for (var i = 5; i * i <= n; i += 6) if (n % i === 0 || n % (i + 2) === 0) return false; return true; }

function applySingleRule(ruleText, totalPoints) {
  var colonIdx = ruleText.indexOf(':');
  if (colonIdx === -1) { statusBar.innerHTML = '❌ 规则格式错误，缺少冒号'; return false; }
  var paramType = ruleText.substring(0, colonIdx).trim();
  var ruleBody = ruleText.substring(colonIdx + 1).trim();
  var paramName = '';
  if (paramType === '速度') paramName = 'speed';
  else if (paramType === '云台俯仰') paramName = 'gimbal';
  else if (paramType === '航向') paramName = 'heading';
  else if (paramType === '高度偏移') paramName = 'altOffset';
  else if (paramType === '云台模式') paramName = 'gimbalMode';
  else if (paramType === '动作类型') paramName = 'action';
  else { statusBar.innerHTML = '❌ 未知参数类型: ' + paramType; return false; }

  // 全局值
  var globalMatch = ruleBody.match(/^(\d+(?:\.\d+)?)$/);
  if (globalMatch) {
    var val = parseFloat(globalMatch[1]);
    for (var i = 0; i < totalPoints; i++) {
      if (paramName === 'speed') waypointParams[i].speed = val;
      else if (paramName === 'gimbal') waypointParams[i].gimbalPitch = val;
      else if (paramName === 'heading') waypointParams[i].heading = val;
      else if (paramName === 'altOffset') waypointParams[i].altOffset = val;
      else if (paramName === 'gimbalMode') waypointParams[i].gimbalMode = val;
      else if (paramName === 'action') waypointParams[i].actionType = val;
    }
    return true;
  }

  // 分段规则 QD-D80=16, D80-ZD=5
  var segments = ruleBody.split(',');
  var segmentApplied = false;
  for (var s = 0; s < segments.length; s++) {
    var seg = segments[s].trim();
    var eqIdx = seg.indexOf('=');
    if (eqIdx === -1) continue;
    var rangePart = seg.substring(0, eqIdx).trim();
    var val = parseFloat(seg.substring(eqIdx + 1).trim());
    if (isNaN(val)) continue;
    var startIdx = 0, endIdx = totalPoints - 1;
    if (rangePart === 'QD') { startIdx = 0; endIdx = 0; }
    else if (rangePart === 'ZD') { startIdx = totalPoints - 1; endIdx = totalPoints - 1; }
    else if (rangePart.indexOf('-') !== -1) {
      var parts = rangePart.split('-');
      var startPart = parts[0].trim(), endPart = parts[1].trim();
      if (startPart === 'QD') startIdx = 0;
      else if (startPart.indexOf('D') === 0) startIdx = parseInt(startPart.substring(1)) - 1;
      else startIdx = parseInt(startPart) - 1;
      if (endPart === 'ZD') endIdx = totalPoints - 1;
      else if (endPart.indexOf('D') === 0) endIdx = parseInt(endPart.substring(1)) - 1;
      else endIdx = parseInt(endPart) - 1;
    } else if (rangePart.indexOf('D') === 0) { var dIdx = parseInt(rangePart.substring(1)) - 1; startIdx = dIdx; endIdx = dIdx; }
    else { var idx = parseInt(rangePart) - 1; startIdx = idx; endIdx = idx; }
    if (startIdx < 0) startIdx = 0;
    if (endIdx >= totalPoints) endIdx = totalPoints - 1;
    for (var i = startIdx; i <= endIdx; i++) {
      if (paramName === 'speed') waypointParams[i].speed = val;
      else if (paramName === 'gimbal') waypointParams[i].gimbalPitch = val;
      else if (paramName === 'heading') waypointParams[i].heading = val;
      else if (paramName === 'altOffset') waypointParams[i].altOffset = val;
      else if (paramName === 'gimbalMode') waypointParams[i].gimbalMode = val;
      else if (paramName === 'action') waypointParams[i].actionType = val;
    }
    segmentApplied = true;
  }
  if (segmentApplied) return true;

  // 奇偶规则
  var oddEvenMatch = ruleBody.match(/奇偶\((\d+(?:\.\d+)?),(\d+(?:\.\d+)?)\)/);
  if (oddEvenMatch) {
    var oddVal = parseFloat(oddEvenMatch[1]), evenVal = parseFloat(oddEvenMatch[2]);
    for (var i = 0; i < totalPoints; i++) { var v = (i % 2 === 0) ? evenVal : oddVal;
      if (paramName === 'speed') waypointParams[i].speed = v;
      else if (paramName === 'gimbal') waypointParams[i].gimbalPitch = v;
      else if (paramName === 'heading') waypointParams[i].heading = v;
      else if (paramName === 'altOffset') waypointParams[i].altOffset = v;
      else if (paramName === 'gimbalMode') waypointParams[i].gimbalMode = v;
      else if (paramName === 'action') waypointParams[i].actionType = v;
    }
    return true;
  }

  // 质数规则
  var primeMatch = ruleBody.match(/质数\((\d+(?:\.\d+)?)\)/);
  if (primeMatch) {
    var primeVal = parseFloat(primeMatch[1]);
    for (var i = 0; i < totalPoints; i++) { if (isPrime(i + 1)) {
      if (paramName === 'speed') waypointParams[i].speed = primeVal;
      else if (paramName === 'gimbal') waypointParams[i].gimbalPitch = primeVal;
      else if (paramName === 'heading') waypointParams[i].heading = primeVal;
      else if (paramName === 'altOffset') waypointParams[i].altOffset = primeVal;
      else if (paramName === 'gimbalMode') waypointParams[i].gimbalMode = primeVal;
      else if (paramName === 'action') waypointParams[i].actionType = primeVal;
    } }
    return true;
  }

  // 正弦波规则
  var sinMatch = ruleBody.match(/正弦波\((\d+(?:\.\d+)?),(\d+(?:\.\d+)?),(\d+(?:\.\d+)?)\)/);
  if (sinMatch) {
    var minVal = parseFloat(sinMatch[1]), maxVal = parseFloat(sinMatch[2]), period = parseFloat(sinMatch[3]);
    if (period <= 0) period = totalPoints;
    for (var i = 0; i < totalPoints; i++) { var t = (i / period) * 2 * Math.PI; var sinVal = minVal + (maxVal - minVal) * (Math.sin(t) + 1) / 2;
      if (paramName === 'speed') waypointParams[i].speed = sinVal;
      else if (paramName === 'gimbal') waypointParams[i].gimbalPitch = sinVal;
      else if (paramName === 'heading') waypointParams[i].heading = sinVal;
      else if (paramName === 'altOffset') waypointParams[i].altOffset = sinVal;
      else if (paramName === 'gimbalMode') waypointParams[i].gimbalMode = Math.round(sinVal);
      else if (paramName === 'action') waypointParams[i].actionType = Math.round(sinVal);
    }
    return true;
  }

  // 斐波那契规则
  var fibMatch = ruleBody.match(/斐波那契\((\d+(?:\.\d+)?),(\d+(?:\.\d+)?)\)/);
  if (fibMatch) {
    var f1 = parseFloat(fibMatch[1]), f2 = parseFloat(fibMatch[2]);
    var fibVals = [f1, f2];
    for (var i = 2; i < totalPoints; i++) { fibVals.push(fibVals[i-1] + fibVals[i-2]); }
    for (var i = 0; i < totalPoints; i++) { var fv = fibVals[i];
      if (paramName === 'speed') waypointParams[i].speed = fv;
      else if (paramName === 'gimbal') waypointParams[i].gimbalPitch = fv;
      else if (paramName === 'heading') waypointParams[i].heading = fv;
      else if (paramName === 'altOffset') waypointParams[i].altOffset = fv;
      else if (paramName === 'gimbalMode') waypointParams[i].gimbalMode = Math.round(fv);
      else if (paramName === 'action') waypointParams[i].actionType = Math.round(fv);
    }
    return true;
  }

  statusBar.innerHTML = '❌ 无法解析规则: ' + ruleBody;
  return false;
}

function applyRules() {
  if (PathManager.pathPoints.length < 2) { statusBar.innerHTML = '❌ 无有效路径点，请先解析路径'; return; }
  initWaypointParams(PathManager.pathPoints.length);
  for (var i = 0; i < pendingRules.length; i++) { applySingleRule(pendingRules[i].param + ':' + pendingRules[i].ruleText, PathManager.pathPoints.length); }
  statusBar.innerHTML = '✅ 已应用 ' + pendingRules.length + ' 条规则，参数已更新';
}

function parseCoordText(text) {
  var tpMatch = text.match(/TP\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/);
  if (tpMatch) { var tx = parseFloat(tpMatch[1]), ty = parseFloat(tpMatch[2]), tz = parseFloat(tpMatch[3]); if (RelCoords.internalCamera) { RelCoords.internalCamera.position.set(tx, ty, tz); RelCoords.internalCamera.lookAt(0,0,0); statusBar.innerHTML='已传送到('+tx.toFixed(1)+','+ty.toFixed(1)+','+tz.toFixed(1)+')'; } return []; }
  var fillMatch = text.match(/\/fill\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+minecraft:(glass|dirt|ice)/i);
  if (fillMatch) { PathManager.executeFill(parseInt(fillMatch[1]), parseInt(fillMatch[2]), parseInt(fillMatch[3]), parseInt(fillMatch[4]), parseInt(fillMatch[5]), parseInt(fillMatch[6]), fillMatch[7].toLowerCase(), RelCoords.scene); return []; }
  var points = [];
  var regex = /(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g;
  var match;
  while ((match = regex.exec(text)) !== null) { points.push({type:match[1], x:parseInt(match[2]), y:parseInt(match[3]), z:parseInt(match[4])}); }
  if (points.length === 0) return [];
  var qd = [], zd = [], dn = [];
  for (var i = 0; i < points.length; i++) { if (points[i].type === 'QD') qd.push(points[i]); else if (points[i].type === 'ZD') zd.push(points[i]); else if (points[i].type.indexOf('D') === 0) dn.push(points[i]); }
  dn.sort(function(a,b){ return parseInt(a.type.substring(1)) - parseInt(b.type.substring(1)); });
  return qd.concat(dn).concat(zd);
}

function createOrbitControls(camera,domElement){ var isDown=false, isRightDrag=false, lastX=0, lastY=0; var target=new THREE.Vector3(0,0,0); var spherical=new THREE.Spherical().setFromVector3(new THREE.Vector3().copy(camera.position).sub(target)); domElement.addEventListener('pointerdown',function(e){ isRightDrag=(e.button===2||e.button===1); isDown=true; lastX=e.clientX; lastY=e.clientY; e.preventDefault(); }); window.addEventListener('pointermove',function(e){ if(!isDown) return; var dx=e.clientX-lastX, dy=e.clientY-lastY; lastX=e.clientX; lastY=e.clientY; if(isRightDrag){ if(RelCoords.lockTranslate) return; var right=new THREE.Vector3().crossVectors(camera.up, new THREE.Vector3().copy(camera.position).normalize()).normalize(); var up=camera.up.clone().normalize(); target.add(right.multiplyScalar(-dx*0.01)).add(up.multiplyScalar(dy*0.01)); } else { if(RelCoords.lockRotate) return; spherical.theta-=dx*0.005; spherical.phi-=dy*0.005; spherical.phi=Math.max(0.1,Math.min(Math.PI-0.1,spherical.phi)); } var offset=new THREE.Vector3().setFromSpherical(spherical); camera.position.copy(target).add(offset); camera.lookAt(target); }); window.addEventListener('pointerup',function(){ isDown=false; isRightDrag=false; }); domElement.addEventListener('wheel',function(e){ if(RelCoords.lockScale) return; spherical.radius*=(e.deltaY>0)?1.05:0.95; spherical.radius=Math.max(0.05,Math.min(100,spherical.radius)); var offset=new THREE.Vector3().setFromSpherical(spherical); camera.position.copy(target).add(offset); camera.lookAt(target); e.preventDefault(); }); domElement.addEventListener('contextmenu',function(e){ e.preventDefault(); }); return {target:target, update:function(){ spherical.setFromVector3(new THREE.Vector3().copy(camera.position).sub(target)); } }; }

function initThreeScene(){
  if(typeof THREE==='undefined'){ setTimeout(initThreeScene,500); return; }
  var scene=new THREE.Scene();
  RelCoords.initScene(scene); FixedFrame.initScene(scene);
  var extPanel=document.getElementById('externalPanel');
  RelCoords.externalRenderer=new THREE.WebGLRenderer({antialias:true,alpha:true});
  RelCoords.externalRenderer.setSize(extPanel.clientWidth,extPanel.clientHeight);
  RelCoords.externalRenderer.setPixelRatio(Math.min(window.devicePixelRatio,2));
  extPanel.appendChild(RelCoords.externalRenderer.domElement);
  RelCoords.externalRenderer.domElement.style.cssText='position:absolute;top:0;left:0;width:100%;height:100%;';
  RelCoords.externalCamera=new THREE.PerspectiveCamera(50, extPanel.clientWidth/extPanel.clientHeight,0.1,100);
  RelCoords.externalCamera.position.set(8,5,8);
  RelCoords.externalCamera.lookAt(0,0,0);
  RelCoords.externalControls=createOrbitControls(RelCoords.externalCamera,RelCoords.externalRenderer.domElement);
  var intPanel=document.getElementById('internalPanel');
  RelCoords.internalRenderer=new THREE.WebGLRenderer({antialias:true,alpha:true});
  RelCoords.internalRenderer.setSize(intPanel.clientWidth,intPanel.clientHeight);
  RelCoords.internalRenderer.setPixelRatio(Math.min(window.devicePixelRatio,2));
  intPanel.appendChild(RelCoords.internalRenderer.domElement);
  RelCoords.internalRenderer.domElement.style.cssText='position:absolute;top:0;left:0;width:100%;height:100%;';
  RelCoords.internalCamera=new THREE.PerspectiveCamera(70, intPanel.clientWidth/intPanel.clientHeight,0.01,100);
  RelCoords.internalCamera.position.set(0,0,5);
  RelCoords.internalCamera.lookAt(0,0,0);
  RelCoords.internalReady=true;
  scene.add(new THREE.AmbientLight(0xffffff,0.6));
  var dl=new THREE.DirectionalLight(0xffffff,0.4); dl.position.set(10,10,10); scene.add(dl);
  FixedFrame.rebuildHollowBox();
  RelCoords.setupInternalControls();
  startRenderLoop('external');
  window.addEventListener('resize',function(){
    if(RelCoords.externalRenderer){ var w=extPanel.clientWidth, h=extPanel.clientHeight; RelCoords.externalRenderer.setSize(w,h); RelCoords.externalCamera.aspect=w/h; RelCoords.externalCamera.updateProjectionMatrix(); }
    if(RelCoords.internalRenderer){ var w=intPanel.clientWidth, h=intPanel.clientHeight; RelCoords.internalRenderer.setSize(w,h); RelCoords.internalCamera.aspect=w/h; RelCoords.internalCamera.updateProjectionMatrix(); }
  });
  statusBar.innerHTML='D3-真实承诺 v1.96 就绪 | 债务驱动规则 + 46列CSV导出';
}

document.getElementById('internalPanel').addEventListener('click',function(e){
  if(RelCoords.internalMode==='observe') return;
  if(!RelCoords.scene||!RelCoords.internalCamera) return;
  var raycaster=new THREE.Raycaster(); var mouse=new THREE.Vector2(); var rect=this.getBoundingClientRect();
  mouse.x=((e.clientX-rect.left)/rect.width)*2-1; mouse.y=-((e.clientY-rect.top)/rect.height)*2+1;
  raycaster.setFromCamera(mouse,RelCoords.internalCamera);
  var allObjects=[]; RelCoords.scene.children.forEach(function(child){ if(child.isMesh && child.userData && child.userData.type){ allObjects.push(child); } });
  if(allObjects.length===0 && RelCoords.internalMode!=='beacon') return;
  var intersects=allObjects.length>0?raycaster.intersectObjects(allObjects,false):[];
  var hitPoint=intersects.length>0?intersects[0].point:null;
  var vc=hitPoint?AbsCoords.worldToVoxel(hitPoint.x,hitPoint.y,hitPoint.z):AbsCoords.worldToVoxel(RelCoords.internalCamera.position.x,RelCoords.internalCamera.position.y,RelCoords.internalCamera.position.z);
  var face=intersects.length>0?intersects[0].face:null; var normalDir=null;
  if(face && face.normal){ var nx=Math.round(face.normal.x), ny=Math.round(face.normal.y), nz=Math.round(face.normal.z); if(nx!==0||ny!==0||nz!==0) normalDir={x:nx,y:ny,z:nz}; }
  if(RelCoords.internalMode==='place'){ PathManager.placeBlock(vc,'path',RelCoords.scene,normalDir); }
  else if(RelCoords.internalMode==='obstacle'){ PathManager.placeBlock(vc,'obstacle',RelCoords.scene,normalDir); }
  else if(RelCoords.internalMode==='remove'){ PathManager.removeBlock(vc,RelCoords.scene); }
  else if(RelCoords.internalMode==='beacon'){ addBeacon(vc); }
  else if(RelCoords.internalMode==='endpoint'){ var key=vc.x+','+vc.y+','+vc.z; PathManager.endpointClickCount[key]=(PathManager.endpointClickCount[key]||0)+1; if(PathManager.endpointClickCount[key]>=3){ PathManager.markAsEndpoint(vc); PathManager.endpointClickCount={}; }else{ statusBar.innerHTML='终点模式：已点击 '+PathManager.endpointClickCount[key]+' 次，需3次确认'; } }
});

window.addEventListener('keydown',function(e){
  if(!document.getElementById('internalPanel').classList.contains('active')) return;
  if(document.activeElement===coordInput) return;
  var mi=document.getElementById('internalMode');
  if(e.key==='1'){ RelCoords.internalMode='place'; mi.textContent='放置模式'; e.preventDefault(); }
  else if(e.key==='2'){ RelCoords.internalMode='obstacle'; mi.textContent='障碍模式'; PathManager.key2PressCount++; if(PathManager.key2PressTimer) clearTimeout(PathManager.key2PressTimer); PathManager.key2PressTimer=setTimeout(function(){ PathManager.key2PressCount=0; },1000); if(PathManager.key2PressCount>=10){ PathManager.clearAllObstacles(RelCoords.scene); PathManager.key2PressCount=0; } e.preventDefault(); }
  else if(e.key==='0'){ RelCoords.internalMode='remove'; mi.textContent='拆除模式'; e.preventDefault(); }
  else if(e.key==='8'){ RelCoords.internalMode='endpoint'; mi.textContent='终点模式'; e.preventDefault(); }
  else if(e.key==='9'){ PathManager.closedLoopMode=!PathManager.closedLoopMode; document.getElementById('toggleClosedLoopBtn').textContent=PathManager.closedLoopMode?'闭环模式':'开环模式'; document.getElementById('toggleClosedLoopBtn').classList.toggle('lock-active',PathManager.closedLoopMode); mi.textContent=PathManager.closedLoopMode?'终点(闭环)':'终点(开环)'; statusBar.innerHTML='已切换为'+(PathManager.closedLoopMode?'闭环':'开环')+'模式'; PathManager.refreshSceneVisuals(RelCoords.scene); e.preventDefault(); }
  else if(e.key==='b'){ RelCoords.internalMode='beacon'; mi.textContent='信标模式'; e.preventDefault(); }
  else if(e.key==='Escape'){ RelCoords.internalMode='observe'; mi.textContent='观察模式'; e.preventDefault(); }
});
