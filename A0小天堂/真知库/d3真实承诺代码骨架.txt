// ==================== 模块0：全局状态管理 ====================
const D3State = {
    // 体素世界参数
    worldSize: { ad: 4, ab: 3, ae: 2.5 },  // 真实物理尺寸（米）
    voxelCount: { x: 22, z: 22, y: 22 },   // 各方向正方体数量
    voxelPhysicalSize: { x: 0, z: 0, y: 0 }, // 每个体素的物理边长（自动计算）

    // 路径数据
    pathPoints: [],  // 路径点数组 [{x, y, z, type: 'QD'|'Dn'|'ZD'}]

    // 锁定状态
    lockTranslate: false,
    lockRotate: false,
    lockScale: false,

    // 外观参数
    opacityFace: 0.35,     // 长方体面不透明度
    opacityGrid: 0.5,      // 体素网格线框不透明度

    // 连接状态
    espConnected: false,
    wifiConnected: false,

    // 初始化
    init() {
        this.calcVoxelSize();
    },

    // 计算每个体素的物理边长
    calcVoxelSize() {
        this.voxelPhysicalSize.x = this.worldSize.ab / this.voxelCount.x;
        this.voxelPhysicalSize.z = this.worldSize.ad / this.voxelCount.z;
        this.voxelPhysicalSize.y = this.worldSize.ae / this.voxelCount.y;
    },

    // 吸附坐标到最近左上角
    snapToCorner(x, y, z) {
        const sx = Math.round(x / this.voxelPhysicalSize.x) * this.voxelPhysicalSize.x;
        const sy = Math.round(y / this.voxelPhysicalSize.y) * this.voxelPhysicalSize.y;
        const sz = Math.round(z / this.voxelPhysicalSize.z) * this.voxelPhysicalSize.z;
        return { x: sx, y: sy, z: sz };
    }
};

// 初始化
D3State.init();

// ==================== 模块6：文本输入与路径管理 ====================
const coordInput = document.getElementById('coordInput');

// 解析坐标文本
function parseCoordText(text) {
    const points = [];
    // 匹配 QD(x,y,z) / D2(x,y,z) / ZD(x,y,z)
    const regex = /(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g;
    let match;
    const matches = [];
    while ((match = regex.exec(text)) !== null) {
        matches.push({
            type: match[1],
            x: parseFloat(match[2]),
            y: parseFloat(match[3]),
            z: parseFloat(match[4])
        });
    }

    // 确保起点QD在最前，终点ZD在最后，中间Dn按数字排序
    const qdPoints = matches.filter(p => p.type === 'QD');
    const zdPoints = matches.filter(p => p.type === 'ZD');
    let dnPoints = matches.filter(p => p.type.startsWith('D') && p.type !== 'QD' && p.type !== 'ZD');
    dnPoints.sort((a, b) => {
        const na = parseInt(a.type.substring(1));
        const nb = parseInt(b.type.substring(1));
        return na - nb;
    });

    // 组装：QD → Dn... → ZD
    const ordered = [...qdPoints, ...dnPoints, ...zdPoints];
    
    // 吸附到左上角坐标
    return ordered.map(p => {
        const snapped = D3State.snapToCorner(p.x, p.y, p.z);
        return { ...p, x: snapped.x, y: snapped.y, z: snapped.z };
    });
}

// 生成指令包
function generatePackage(points) {
    if (points.length < 2) return null;

    let output = '';
    const pathTypes = [];

    // 判断路径类型（简化版）
    if (points.length === 2) {
        pathTypes.push('直线路径');
    } else if (points.length > 2) {
        pathTypes.push('自由路径');
    }

    output += `路径类型：${pathTypes.join(' + ')}\n`;
    output += `总段数：${points.length - 1}\n`;
    output += `\n路径坐标清单：\n`;
    
    const coordList = points.map(p => `${p.type}(${p.x},${p.y},${p.z})`).join(' → ');
    output += coordList + '\n\n';

    // 计算每段leg
    for (let i = 0; i < points.length - 1; i++) {
        const a = points[i];
        const b = points[i + 1];
        const legX = Math.abs(b.x - a.x);
        const legZ = Math.abs(b.z - a.z);
        const legY = Math.abs(b.y - a.y);
        output += `段${i + 1}: legX=${legX.toFixed(2)}m, legZ=${legZ.toFixed(2)}m, legY=${legY.toFixed(2)}m\n`;
    }

    return output;
}

// 解析按钮点击
parseBtn.addEventListener('click', () => {
    const text = coordInput.value.trim();
    if (!text) {
        statusBar.innerHTML = '⚠️ 请输入坐标指令';
        return;
    }

    const points = parseCoordText(text);
    if (points.length === 0) {
        statusBar.innerHTML = '❌ 未能识别坐标格式，请检查';
        return;
    }

    // 检查是否有起点和终点
    const hasQD = points.some(p => p.type === 'QD');
    const hasZD = points.some(p => p.type === 'ZD');
    if (!hasQD || !hasZD) {
        statusBar.innerHTML = '⚠️ 缺少起点(QD)或终点(ZD)，请补充';
        return;
    }

    // 存入全局状态
    D3State.pathPoints = points;
    statusBar.innerHTML = `✅ 路径已更新 | ${points.length}个坐标点 | 起点${points[0].type}(${points[0].x},${points[0].y},${points[0].z}) → 终点${points[points.length-1].type}(${points[points.length-1].x},${points[points.length-1].y},${points[points.length-1].z})`;
    outputArea.textContent = '路径已更新，点击"生成指令包"查看完整指令。';
});

// 生成指令包按钮
document.getElementById('generatePkgBtn').addEventListener('click', () => {
    if (D3State.pathPoints.length < 2) {
        outputArea.textContent = '❌ 路径点不足，请先解析坐标';
        return;
    }
    const pkg = generatePackage(D3State.pathPoints);
    if (pkg) {
        outputArea.textContent = pkg;
        statusBar.innerHTML = '📦 指令包已生成';
    }
});

// 复制指令包
document.getElementById('copyPkgBtn').addEventListener('click', () => {
    const text = outputArea.textContent;
    if (text && text !== '等待输入坐标...' && !text.startsWith('❌') && !text.startsWith('路径已更新')) {
        navigator.clipboard.writeText(text).then(() => {
            statusBar.innerHTML = '📋 指令包已复制到剪贴板';
        }).catch(() => {
            statusBar.innerHTML = '❌ 复制失败，请手动复制';
        });
    } else {
        statusBar.innerHTML = '⚠️ 请先生成指令包';
    }
});
