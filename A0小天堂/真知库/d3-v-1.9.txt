    if (tpMatch) {
        var tx=parseFloat(tpMatch[1]), ty=parseFloat(tpMatch[2]), tz=parseFloat(tpMatch[3]);

        return [];
    }
    var fillMatch = text.match(/\/fill\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+(-?\d+)\s+minecraft:(glass|dirt|ice)/i);
    if (fillMatch) {
        var x1=parseInt(fillMatch[1]),y1=parseInt(fillMatch[2]),z1=parseInt(fillMatch[3]),x2=parseInt(fillMatch[4]),y2=parseInt(fillMatch[5]),z2=parseInt(fillMatch[6]),blockType=fillMatch[7].toLowerCase();
        var minX=Math.min(x1,x2),maxX=Math.max(x1,x2),minY=Math.min(y1,y2),maxY=Math.max(y1,y2),minZ=Math.min(z1,z2),maxZ=Math.max(z1,z2);
        var count=0;
        if(blockType==='glass'){
            for(var vx=minX;vx<=maxX;vx++){for(var vy=minY;vy<=maxY;vy++){for(var vz=minZ;vz<=maxZ;vz++){addManualPoint({x:vx,y:vy,z:vz});count++;}}}
            statusBar.innerHTML='Fill指令：生成'+count+'个玻璃路径方块';
        }else if(blockType==='dirt'){
            if(!D3State.scene)return[];
            for(var vx=minX;vx<=maxX;vx++){for(var vy=minY;vy<=maxY;vy++){for(var vz=minZ;vz<=maxZ;vz++){
                var rc=D3State.getVoxelRenderCenter({x:vx,y:vy,z:vz});var geo=new THREE.BoxGeometry(D3State.voxelRenderSize*0.95,D3State.voxelRenderSize*0.95,D3State.voxelRenderSize*0.95);

    }
    var points=[];var regex=/(QD|D\d+|ZD)\((-?[\d.]+),(-?[\d.]+),(-?[\d.]+)\)/g;var match;
    while((match=regex.exec(text))!==null){points.push({type:match[1],x:parseInt(match[2]),y:parseInt(match[3]),z:parseInt(match[4])});}
    if(points.length===0)return[];

    dn.sort(function(a,b){return parseInt(a.type.substring(1))-parseInt(b.type.substring(1));});

function generatePackage(points){
    if(points.length<2)return null;
    var out='路径类型：'+(points.length===2?'直线':'自由')+'路径'+(D3State.closedLoopMode?'（闭环）':'（开环）')+'\n总段数：'+(points.length-1)+'\n\n路径坐标清单（体素）：\n';
    for(var i=0;i<points.length;i++)out+=points[i].type+'('+points[i].x+','+points[i].y+','+points[i].z+') → ';
    out=out.slice(0,-3)+'\n\n物理距离清单：\n';var mpv=D3State.meterAB/D3State.voxelCountAB;
    for(var j=0;j<points.length-1;j++){var a=points[j],b=points[j+1];out+='段'+(j+1)+': legX='+(Math.abs(b.x-a.x)*mpv).toFixed(2)+'m, legZ='+(Math.abs(b.z-a.z)*mpv).toFixed(2)+'m, legY='+(Math.abs(b.y-a.y)*mpv).toFixed(2)+'m\n';}
    return out;
}

function generateSequence(points){
    if(points.length<2)return null;
    var out='时序指令清单：\n';var mpv=D3State.meterAB/D3State.voxelCountAB;var currentDir='';
    for(var j=0;j<points.length-1;j++){
        var a=points[j],b=points[j+1];var dx=b.x-a.x,dz=b.z-a.z,dy=b.y-a.y;
        var distX=Math.abs(dx)*mpv,distZ=Math.abs(dz)*mpv,distY=Math.abs(dy)*mpv;
        var dir='';
        if(dx>0)dir='东';else if(dx<0)dir='西';
        if(dz>0)dir+='南';else if(dz<0)dir+='北';
        if(dir==='')dir='原地';
        var turn='';
        if(currentDir!==''&&currentDir!==dir){
            var angleMap={'东':'90°','西':'90°','南':'90°','北':'90°'};
            turn='右转 90° | ';
        }
        currentDir=dir;
        var totalDist=Math.sqrt(distX*distX+distZ*distZ+distY*distY);
        out+=(j+1)+'. '+turn+'前进 '+totalDist.toFixed(2)+'m（向'+dir+'）\n';
    }
    out+=points.length+'. 停止';
    return out;
}

function updatePathLineInScene(){
    if(!D3State.scene)return;
    if(D3State.pathLine){D3State.scene.remove(D3State.pathLine);D3State.pathLine=null;}
    if(D3State.arrowMesh){D3State.scene.remove(D3State.arrowMesh);D3State.arrowMesh=null;}
    if(D3State.pathPoints.length<2)return;
    var pts=[];
    for(var i=0;i<D3State.pathPoints.length;i++){var rc=D3State.getVoxelRenderCenter(D3State.pathPoints[i]);pts.push(new THREE.Vector3(rc.x,rc.y,rc.z));}
    // 闭环模式：最后一个点连回第一个点
    if(D3State.closedLoopMode&&D3State.pathPoints.length>=2){
        var firstRc=D3State.getVoxelRenderCenter(D3State.pathPoints[0]);
        pts.push(new THREE.Vector3(firstRc.x,firstRc.y,firstRc.z));
    }
    var geo=new THREE.BufferGeometry().setFromPoints(pts);
    D3State.pathLine=new THREE.Line(geo,new THREE.LineBasicMaterial({color:0xff3333,linewidth:2}));D3State.scene.add(D3State.pathLine);
    var last=pts[pts.length-1],prev=pts[pts.length-2]||pts[0];
    var dir=new THREE.Vector3().subVectors(last,prev).normalize();if(dir.length()<0.001)dir.set(0,0,1);
    var arrowMesh=new THREE.Mesh(new THREE.ConeGeometry(0.2,0.5,8,8),new THREE.MeshBasicMaterial({color:0xff3333}));
    arrowMesh.position.copy(last);arrowMesh.quaternion.setFromUnitVectors(new THREE.Vector3(0,1,0),dir);
    D3State.arrowMesh=arrowMesh;D3State.scene.add(arrowMesh);
}
